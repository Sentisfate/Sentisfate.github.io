解法1 冒泡排序

思路：

将大值冒泡到后面

临界条件： :::info

::: 证明： :::tips

:::

/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var sortColors = function(nums) {
    // 使用冒泡排序
    let len=nums.length
    for(let i=0;i<len;i++){
        let min
        for(let j=1;j<len-i;j++){
            if(nums[j-1]>nums[j]){
                min=nums[j]
                nums[j]=nums[j-1]
                nums[j-1]=min
            }
        }
    }
};

解法2 两次遍历

思路：

记录0，1，2的个数，直接修改数组的值

临界条件： :::info

::: 证明： :::tips

:::

/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var sortColors = function(nums) {
    let len=nums.length
    //记录0，1，2的个数为a，b，c
    let a=b=c=0
    for(let i=0;i<len;i++){
        switch(nums[i]){
            case 0:a++
            break;
            case 1:b++
            break;
            default: c++
        }
    }
    for(let i=0;i<len;i++){
        if(i<a){
            nums[i]=0
        }else if(i<b+a){
            nums[i]=1
        }else{
            nums[i]=2
        }
    }
};

解法3一次遍历

思路：

用两个指针来标识头部和尾部的位置，在遍历的过程中，将0移动到头部，将1移动到尾部。

/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var sortColors = function(nums) {
    let len=nums.length
    let head=0
    let tail=len-1
    const swap=(nums,i,j)=>{
        let temp=nums[i]
        nums[i]=nums[j]
        nums[j]=temp
    }
    for(let i=0;i<=tail;i++){
        //当nums[i]为2时，就移动到后面，并且继续判断交换后的是否为2，
        while(i<=tail&&nums[i]==2){
            swap(nums,i,tail--)
        }
        if(nums[i]==0){
            swap(nums,i,head++)
        }
    }
};
