  • 贡献度取决于包含该元素的且该元素为子数组的最小值的个数。

  1. 栈为空时入栈
  2. 不为空，比较入栈元素和栈顶元素，如果入栈元素大于栈顶元素，则继续入栈，否则，出栈，出栈的元素符合条件的最大子数组即可以确定为栈顶元素和入栈元素之间的所有元素。计算公式为：

  3. 重复上述过程，直至遍历完所有的元素，再依次将元素出栈，并计算其贡献度，

/**
 * @param {number[]} arr
 * @return {number}
 */
var sumSubarrayMins = function(arr) {
  const s=[]
  let mod=10**9+7
  let len=arr.length
  let sum=BigInt(0)
  for(let i=0;i<len;i++){
    if(s.length==0){
      s.push(i)
      continue
    }
    let top=s[s.length-1]
    if(arr[i]>arr[top]){
      s.push(i)
      continue
    }
    while(arr[i]<=arr[top]&&s.length){
      let index=s.pop()
      if(s.length) top=s[s.length-1]
      else top=-1
      sum+=BigInt(index-top)*BigInt(i-index)*BigInt(arr[index])
    }
    s.push(i)
  }
  while(s.length){
    let index=s.pop()
    let top
    if(s.length) top=s[s.length-1]
    else top=-1
    sum+=BigInt(index-top)*BigInt(len-index)*BigInt(arr[index])
  }
  return sum%BigInt(mod)
};
