两正序数组的中位数

2022-08-22 14:00

解法1，合并排序后找中位数

  • 时间复杂度 O((n+m)2)
  • 空间复杂度 O(2m+2n)
function getMediumNumByMerge() {
  const arr = [...nums1, ...nums2];
  let m = nums1.length;
  let n = nums2.length;
  arr.sort((a, b) => a - b);
  let medium = (m + n) / 2;
  if ((m + n) % 2) {
    medium = Math.floor(medium);
    return arr[medium];
  }
  return (arr[medium] + arr[medium - 1]) / 2;
}

解法2逐个比较查找中位数的位置

function getMediumNumByLoop(nums1, nums2) {
  let m = nums1.length;
  let n = nums2.length;
  if ((m == 0 && n == 1) || (n == 0 && m == 1)) {
    return nums1[0] === undefined ? nums2[0] : nums1[0];
  }
  let i = 0;
  let j = 0;
  let medium = Math.floor((m + n) / 2 + 1);
  let arr = [];
  while (i + j + 2 <= medium + 1) {
    if (i < m && j < n) {
      arr.push(nums1[i] < nums2[j] ? nums1[i++] : nums2[j++]);
      continue;
    }
    if (i < m) {
      arr.push(nums1[i++]);
      continue;
    }
    arr.push(nums2[j++]);
  }

  if ((m + n) % 2 == 0) {
    return (arr[arr.length - 1] + arr[arr.length - 2]) / 2;
  }
  return arr[arr.length - 1];
}

解法3：二分查找

核心思想：中位数是排序后的第(m+n)/2个元素，或者(m+n)/2与(m+n)/2+1的平均值，总的来说是第k小个数相较于解法2的逐个排查法，可以一次性排查一半k/2来快速排查。

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findMedianSortedArrays = function(nums1, nums2) {
    // 获得第k个元素
    const getKthElement=(nums1,nums2,k)=>{
        let m=nums1.length,n=nums2.length
        //当数组1或2长度为0时，就返回另一数组的第k个数字
        if(m==0){
            return nums2[k-1]
        }
        if(n==0){
            return nums1[k-1]
        }
        if(k<2){
            console.log(k,nums1[0],nums2[0])
            return Math.min(nums1[0],nums2[0])
        }
        //注意要保证一半的长度要在两数组长度之内
        let half = Math.min(Math.floor(k / 2), m, n);
        k -= half;
        if(nums1[half-1]<=nums2[half-1]){
            // 获得剩下数组的第k个
            return getKthElement(nums1.slice(half),nums2,k)
        }
        return getKthElement(nums1,nums2.slice(half),k)
    }
    let m=nums1.length,n=nums2.length
    // 为奇数，k就为m+n /2 +1
    if((m+n)%2==1){
        let k=Math.ceil((m+n)/2)
        return getKthElement(nums1,nums2,k)
    }
    // 为偶数
    let k1=(m+n)/2
    let k2=k1+1
    return getKthElement(nums1,nums2,k1)*0.5+getKthElement(nums1,nums2,k2)*0.5
    // 获得第k个元素
};
