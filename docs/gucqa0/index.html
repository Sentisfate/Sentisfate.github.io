雇佣k名工人的最低成本

2022-09-11 05:32

题目描述

解法1 排序+枚举

思路：

每个工人都有可能成为其中的一员
多名工人的工资应如何发？
对于每个工人来说要尽可能少的发工资，也就是发最低期望的工资
按照该工资与质量之比来给其他工人发工资，看看能否满足其他人，
如果满足了，则可以发工资给他们，他们可以作为人选，否则就换成下一个人以最低期望的工资

临界条件： :::info

::: 证明： :::tips

:::

/**
 * @param {number[]} quality
 * @param {number[]} wage
 * @param {number} k
 * @return {number}
 */
var mincostToHireWorkers = function(quality, wage, k) {
    // 每个工人都有可能成为其中的一员
    // 多名工人的工资应如何发
    // 对于每个工人来说要尽可能少的发工资，也就是发最低期望的工资
    // 按照该工资与质量之比来给其他工人发工资，看看能否满足其他人，
    // 如果满足了，则可以发工资给他们，他们可以作为人选，否则就换成下一个人以最低期望的工资
    let copy=wage.map((item,i)=>[quality[i],item])
    //按照质量进行一个升序排列
    copy.sort((a,b)=>a[0]-b[0])
    console.log(copy)
    let minSalary=0
    let len=quality.length
    for(let i=0;i<len;i++){
        //实际工资
        let salaries=[copy[i][1]] 
        //wage/quality
        let ratio=copy[i][1]/copy[i][0]
        let j=0
        while(salaries.length<k&&j<len){
            if(j==i){
                j++
                continue
            }
            let salary=copy[j][0]*ratio
            if(salary>=copy[j][1]){
                salaries.push(salary)
            }
            j++
        }
        if(salaries.length<k){
            continue
        }
        //如果此种情况的人选大于等于k的话，就可以从中选出最小的工资总和
        let sum=salaries[0]
        for(let i=1;i<k;i++){
            sum+=salaries[i]
        }
        if(minSalary>sum||minSalary==0){
            minSalary=sum
        }
    }
    return minSalary
};

故，最少工资应该由”性价比“最大的那个人决定。故我们只需要关注于性价比即可，在工作组中的k人应该保证，其余k-1人的性价比小于第k人 :::

/**
 * @param {number[]} quality
 * @param {number[]} wage
 * @param {number} k
 * @return {number}
 */
var mincostToHireWorkers = function(quality, wage, k) {
    let wqs=wage.map((item,i)=>[item,quality[i]])
    //将性价比从低到高排列
    wqs.sort((a,b)=>a[0]/a[1]-b[0]/b[1])
    //第k个开始的工人能够作为工作组中性价比最大的那个人
    let len=wage.length
    const getSum=(arr)=>{
        let sum=0
        for(let i=0;i<arr.length;i++){
            sum+=arr[i]
        }
        return sum
    }
    //将前k-1个人按照工作质量排序，作为一个初始队列
    let qs=wqs.map(item=>item[1]).slice(0,k-1).sort((a,b)=>a-b)
    let min=getSum(qs)*wqs[k-1][0]/wqs[k-1][1]+wqs[k-1][0]
    for(let i=k;i<len;i++){
        let q=wqs[i][1]
        let w=wqs[i][0]
        let ratio=w/q
        //总工资为总质量*ration
        //因此希望人数一定的情况下，总工作质量越小越好
        // 将第i-1个人插入到队列中，保持队列长度为k-1即可
        for(let j=0;j<qs.length;j++){
            if(wqs[i-1][1]<qs[j]){
                //插入元素
                qs.splice(j,0,wqs[i-1][1])
                break;
            }
        }
        qs.length=k-1
        min=Math.min(min,getSum(qs)*ratio+w)
    }
    return min
};
