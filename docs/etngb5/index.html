在LR字符串中交换相邻字符

2022-10-02 03:17

题目描述

解法1

思路：

对于题目中描述，L只能向左移动，R只能向右移动，移动的前提是，与之相邻的是X，且移动后相对位置不会发生变化，此外，注意到一点由于L只能向右移动，start中的L必须在end中的对应的L的右边。R则必须在左边。

临界条件： :::info

::: 证明： :::tips

:::

/**
 * @param {string} start
 * @param {string} end
 * @return {boolean}
 */
var canTransform = function(start, end) {
  let len=start.length
  if(len!=end.length||(len==1&&start!=end))return false
  let i=j=0
  while(i<len&&j<len){
    // 逐个比较L，和R，
    if(start[i]=='X') {
      i++
      continue
    }
    
    if(end[j]=='X') {
      j++
      continue
    }
    // 此时比较start[i]和end[j]是否相等，不相等返回false
    // 如果是L的话 i<=j如果是R的话，i>=j
    if(start[i]!=end[j]||start[i]=='L'&&i<j||start[i]=='R'&&i>j) return false
    i++
    j++
  }
  console.log(i,j)
  // 统计i或者j后面的字符是否有R或L，如果有的话说明R，L的个数不相等
  for(let index=i;i<len;i++){
    if(start[i]!='X') return false
  }
  for(let index=j;j<len;j++){
    if(end[j]!='X') return false
  }
  return true
};
