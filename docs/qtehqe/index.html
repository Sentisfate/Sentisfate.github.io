940不同的子序列

/**
 * @param {string} s
 * @return {number}
 */
var distinctSubseqII = function(s) {
  //对于字符串s，使用dp[i]来存储以前i+1个字符组成的子序列的个数。
  //对于dp[i]而言，子序列相较于dp[i-1]多了一个以s[i]结尾的子序列集合，如果s[i]没有重复的话，则子序列集合可以为dp[i-1]所有子序列后加上s[i]组成的子序列以及单个s[i]，即dp[i-1]+1，记为newCount
  // 如果有s[i]在前面有重复，以aba 为例，发现在newCount中出现了重复，重复的内容是上个a新增的个数，记重复的个数为repeatCount
  // dp[i]=dp[i-1]+newCount-repeatCount
  const mod=10**9+7
  // 记录每个字母结尾的子序列新增的个数
  const preCount=new Array(26).fill(0)
  // 用两个变量来记录dp[i]和dp[i-1]
  //没开始遍历前相当于是一个''
  let cur=1
  for(let i=0;i<s.length;i++){
    //当前字符在数组的索引
    let index=s.charCodeAt(i)-'a'.charCodeAt(0)
    //记录新增的个数
    let newCount=cur
    //重复的个数
    let repeatCount=preCount[index]%mod
    //记录下当前字符的新增值
    preCount[index]=newCount
    cur=(((cur+newCount)%mod)-repeatCount+mod)%mod
  }
  //去除掉''
  return cur-1
};
