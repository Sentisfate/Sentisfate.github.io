  static getSiteTime(trackCate3: Cartesian3[], speed = 25) {
    let timeSum = 0;
    let siteTimes = [0];

    for (let i = 1; i < trackCate3.length; i++) {
      let prePoint = trackCate3[i - 1];
      let curPoint = trackCate3[i];
      timeSum += TracksManager.getSpaceDistance(prePoint, curPoint) / speed;
      siteTimes.push(timeSum);
    }
    return { timeSum, siteTimes };
  }
  // è®¡ç®—å¤šä¸ªç‚¹çš„æ€»è·ç¦»
  static getSpaceDistance(...positions: Cartesian3[]): number {
    let sumDistance = 0;
    for (let i = 1; i < positions.length; i++) {
      let prePoint = Cartographic.fromCartesian(positions[i - 1]);
      let curPoint = Cartographic.fromCartesian(positions[i]);
      // è®¡ç®—å½“å‰ç‚¹åˆ°ä¸Šä¸€ç‚¹çš„è·ç¦»
      const geodesic = new EllipsoidGeodesic();
      geodesic.setEndPoints(prePoint, curPoint);
      const horizontalDistance = geodesic.surfaceDistance;
      const verticalDistance = prePoint.height - curPoint.height;
      sumDistance += Math.sqrt(
        Math.pow(horizontalDistance, 2) + Math.pow(verticalDistance, 2)
      );
    }
    return sumDistance;
  }

  getTracksWithJulidateFromTracks(tracks: Track[]) {
    const tracksWithJuliDate: TrackWithJuliDate[] = [];
    this.startTime = JulianDate.fromDate(new Date());
    let stopTime = this.startTime.clone();
    // å¦‚æœæœ‰æ—¶é—´ï¼Œæ ¹æ®æ—¶é—´è®¡ç®—é€Ÿåº¦
    if (this.times) {
      this.speeds = [];

        let speed = TracksManager.getSpaceDistance(...tracks[i]) / time;
        this.speeds.push(speed);
      });
    }
    // æŒ‰ç…§é€Ÿåº¦è®¡ç®—æ—¥æœŸ
    tracks.forEach((track, i) => {
      let { siteTimes, timeSum } = TracksManager.getSiteTime(
        track,
        this.speeds[i]

      tracksWithJuliDate.push([]);
      siteTimes.forEach((time, j) => {
        tracksWithJuliDate[i].push({
          position: track[j],
          time: JulianDate.addSeconds(stopTime, time, new JulianDate()),
        });
      });
      stopTime = JulianDate.addSeconds(stopTime, timeSum, new JulianDate());
    });

    this.stopTime = stopTime;
    return tracksWithJuliDate;

  getStartAndStopFromTracksWithTime(tracks: TrackWithTime[]) {
    let minStart = tracks[0][0].time;
    let maxEnd = tracks[0][tracks[0].length - 1].time;
    for (let i = 1; i < tracks.length; i++) {
      let start = tracks[i][0].time;
      let end = tracks[i][tracks[i].length - 1].time;
      if (minStart > start) minStart = start;
      if (maxEnd < end) maxEnd = end;
    }
    this.startTime = JulianDate.fromDate(minStart);
    this.stopTime = JulianDate.fromDate(maxEnd);
    return [minStart, maxEnd];
  }
  getStartAndStopFromTracksWithJulidate(tracks: TrackWithJuliDate[]) {
    if (tracks.length == 0) {
      this.startTime = new JulianDate();
      this.stopTime = new JulianDate();
    }
    let minStart = tracks[0][0].time;
    let maxEnd = tracks[0][tracks[0].length - 1].time;
    for (let i = 1; i < tracks.length; i++) {
      let start = tracks[i][0].time;
      let end = tracks[i][tracks[i].length - 1].time;
      if (JulianDate.compare(minStart, start) > 0) minStart = start;
      if (JulianDate.compare(maxEnd, end) < 0) maxEnd = end;
    }
    this.startTime = minStart;
    this.stopTime = maxEnd;
    return [minStart, maxEnd];
  }

  fillArray<T>(value: T, length?: number) {
    if (!length) return [];
    const arr = [];
    for (let i = 0; i < length; i++) {
      arr.push(value);

  getTwoDimensionArray<T1, T2>(

      return [arr] as Array<T1> | Array<T2>;

  removeItemInArr<T>(arr: Array<T>, item: T): Array<T> {
    const newArr: T[] = [];
    for (let ele of arr) {
      if (item === arr) continue;
      newArr.push(ele);
    }
    return newArr;
  }
