import {
  Viewer,
  Cartesian3,
  Cartographic,
  EllipsoidGeodesic,
  TimeIntervalCollection,
  TimeInterval,
  Color,
  JulianDate,
  SampledPositionProperty,
  VelocityOrientationProperty,
  Entity,
  ModelGraphics,
} from 'cesium';

type Track = Cartesian3[];
type TrackWithTime = Array<{ position: Cartesian3; time: Date }>;
type TrackWithJuliDate = Array<{
  position: Cartesian3;
  time: JulianDate;
}>;
type Tracks = Track | Track[] | TrackWithTime | TrackWithTime[];
type ModelUrl = string[] | string;
type Speeds = number | number[]; //每条轨迹的速度，当轨迹无时间时，可提供一个速度
type Times = number[]; //每条轨迹所用时长，与速度只能二选一，同时存在以时间为准
type TrackOption =
  | {
      tracks: Tracks;
      modelUrl: ModelUrl;
      speeds?: Speeds;
      times?: Times;
      tracksWithJulidate?: TrackWithJuliDate[] | TrackWithJuliDate;
    }
  | {
      tracks?: Tracks;
      modelUrl: ModelUrl;
      speeds?: Speeds;
      times?: Times;
      tracksWithJulidate: TrackWithJuliDate[] | TrackWithJuliDate;
    };
interface CreateEntityOption {
  startTime: JulianDate;
  stopTime: JulianDate;
  property: SampledPositionProperty;
  modelUrl: string;
  options?: Entity.ConstructorOptions;
}

// 大概的功能
/**
 * 默认初始化时，而已初始化一些轨迹，并进行管理
 * 允许添加删除轨迹
 */
function lengthIsNotSame() {
  throw new Error('轨迹与times,speeds长度不一致');
}

export class TracksManager {
  viewer: Viewer;
  entities: Entity[];
  // tracks?: Track[] | TrackWithTime[]; //轨迹集合实例化时传入
  tracksWithJulidate!: TrackWithJuliDate[]; //可传入带有儒略日的轨迹
  modelUrls: string[]; //每条轨迹的模型，如果只有一个，则所有的都是该模型
  speeds: number[] = []; //每条轨迹的速度，m/s 当与times同时出现时，以times为准
  times?: Times; //每��轨迹所用的时间 秒
  startTime!: JulianDate; //所有轨迹中最先开始的时候
  stopTime!: JulianDate; //最后结束的时候
  trackEntityMap: WeakMap<TrackWithJuliDate, Entity> = new WeakMap();
  // [key: string]: any;
  constructor(viewer: Viewer, options: TrackOption) {
    this.viewer = viewer;
    this.entities = [];
    if (options.speeds) {
      if (Array.isArray(options.speeds)) {
        this.speeds = options.speeds;
      } else
        this.speeds = this.fillArray(
          options.speeds,
          options.tracks?.length || options.tracksWithJulidate?.length
        );
    }
    this.times = options.times;
    // 处理路径为 TracksWithJulidate
    if (options.tracksWithJulidate) {
      this.tracksWithJulidate = this.getTwoDimensionArray<
        TrackWithJuliDate,
        TrackWithJuliDate
      >(options.tracksWithJulidate);
      this.getStartAndStopFromTracksWithJulidate(this.tracksWithJulidate);
    } else if (options.tracks) {
      const tracks = this.getTwoDimensionArray<Track, TrackWithTime>(
        options.tracks
      );
      if (options.speeds && tracks.length !== this.speeds.length)
        lengthIsNotSame();
      if (options.times && options.times.length !== tracks.length)
        lengthIsNotSame();
      this.tracksWithJulidate = this.toTracksWithJuliDate(tracks);
    }
    // 构建modelUrl
    if (Array.isArray(options.modelUrl)) this.modelUrls = options.modelUrl;
    else
      this.modelUrls = this.fillArray(
        options.modelUrl,
        this.tracksWithJulidate.length
      );
    // 创建实体
    this.createEntities();
  }
  /**
   * 播放1个或所有的轨迹
   * @param index 要回放的轨迹或轨迹索引,如果没有赋值，则回放所有的轨迹
   * @returns
   */
  public playOne(track: number | TrackWithJuliDate) {
    let entity: Entity;
    let twj: TrackWithJuliDate;
    if (typeof track === 'number') {
      let index = track;
      twj = this.tracksWithJulidate[index];
    } else {
      twj = track;
    }
    if (this.trackEntityMap.has(twj)) {
      entity = this.trackEntityMap.get(twj) as Entity;
    } else throw new Error('没有管理该轨迹');
    this.addEntity(entity);
    this.setClockTime(twj[0].time, twj[twj.length - 1].time);
    return entity;
  }
  setClockTime(start: JulianDate, stop: JulianDate) {
    this.viewer.clock.startTime = start;
    this.viewer.clock.stopTime = stop;
    this.viewer.clock.shouldAnimate = true;
  }
  play(tracks?: number[] | TrackWithJuliDate[]) {
    const entities: Entity[] = [];
    let twjs: TrackWithJuliDate[] = [];
    if (typeof tracks === 'undefined' || tracks.length === 0) {
      // 播放所有的
      this.entities.forEach((entity) => {
        this.viewer.entities.add(entity);
      });
      this.setClockTime(this.startTime, this.stopTime);
      return this.entities;
    }
    if (typeof tracks[0] === 'number') {
      tracks.forEach((index) => {
        const twj = this.tracksWithJulidate[index as number];
        twjs.push(twj);
        if (!this.trackEntityMap.has(twj))
          throw new Error('没有找到该轨迹的实体');
        const e = this.trackEntityMap.get(twj) as Entity;
        entities.push(e);
        this.viewer.entities.add(e);
      });
    } else {
      twjs = tracks as TrackWithJuliDate[];
      twjs.forEach((twj) => {
        if (!this.trackEntityMap.has(twj))
          throw new Error('没有找到该轨迹的实体');
        const e = this.trackEntityMap.get(twj) as Entity;
        entities.push(e);
        this.viewer.entities.add(e);
      });
    }
    // 计算开始结束时间
    const [start, stop] = this.getStartAndStopFromTracksWithJulidate(twjs);
    this.setClockTime(start, stop);
    return entities;
  }
  addEntity(entity: Entity) {
    if (this.viewer.entities.contains(entity)) return;
    this.viewer.entities.add(entity);
  }
  /**
   * 从viewer中移除指定的实体
   * @param entity 要移除的实体
   */
  removeEntity(entity: Entity) {
    if (this.viewer.entities.contains(entity))
      this.viewer.entities.remove(entity);
  }
  /**
   * 从viewer中移除所有实体
   */
  removeAll() {
    for (let entity of this.entities) {
      this.removeEntity(entity);
    }
  }
  /**
   * 添加一条新的轨迹
   * @param trackOption
   *
   */
  addTrack(trackOption: {
    track: Track | TrackWithTime;
    speed?: number;
    time?: number;
    modelUrl: string;
    startTime?: JulianDate;
    entityOption?: Entity.ConstructorOptions;
  }) {
    // 转换为trackWithJulidate
    const { track, speed, time, modelUrl, startTime, entityOption } =
      trackOption;
    let twj: TrackWithJuliDate;
    if (track[0] instanceof Cartesian3) {
      twj = this.getTWJFromTrack(
        track as Track,
        startTime || JulianDate.fromDate(new Date()),
        speed,
        time
      );
    } else {
      twj = this.getTWJFromTWT(track as TrackWithTime);
    }
    this.addTrackWithJulidate({ track: twj, modelUrl: modelUrl, entityOption });
  }

  removeTrack(track?: TrackWithJuliDate) {
    if (track) {
      if (!this.trackEntityMap.has(track)) return;
      // 删除实体
      const e = this.trackEntityMap.get(track) as Entity;
      this.entities = this.removeItemInArr(this.entities, e);
      // 移除viewer的实体
      this.removeEntity(e);
      // 删除映射
      this.trackEntityMap.delete(track);
      // 删除轨迹
      this.tracksWithJulidate = this.removeItemInArr(
        this.tracksWithJulidate,
        track
      );
    } else {
      // 移除所有
      this.removeAll();
      this.entities = [];
      while (this.tracksWithJulidate.length) {
        const twj = this.tracksWithJulidate.pop() as TrackWithJuliDate;
        this.trackEntityMap.delete(twj);
      }
    }
    // 重新确定起止时间
    this.getStartAndStopFromTracksWithJulidate(this.tracksWithJulidate);
  }
  /**
   *
   * @param trackOption 要添加实体的选项
   */
  addTrackWithJulidate(trackOption: {
    track: TrackWithJuliDate;
    modelUrl: string;
    entityOption?: Entity.ConstructorOptions;
  }) {
    this._addTWJ(trackOption);
    // 更新整个manager的startTime和stopTime
    this.getStartAndStopFromTracksWithJulidate(this.tracksWithJulidate);
  }
  private _addTWJ(trackOption: {
    track: TrackWithJuliDate;
    modelUrl: string;
    entityOption?: Entity.ConstructorOptions;
  }) {
    this.modelUrls.push(trackOption.modelUrl);
    this.tracksWithJulidate.push(trackOption.track);
    this.createEntities(
      this.tracksWithJulidate.length - 1,
      trackOption.entityOption
    );
  }
  addTracksWithJulidate(
    trackOptions: Array<{
      track: TrackWithJuliDate;
      modelUrl: string;
      entityOption?: Entity.ConstructorOptions;
    }>
  ) {
    trackOptions.forEach((trackOption) => {
      this._addTWJ(trackOption);
    });
    this.getStartAndStopFromTracksWithJulidate(this.tracksWithJulidate);
  }
  // deleteTrack(track: TrackWithJuliDate) {
  //   // this.removeItemInArr()
  // }

  // 移除回放过路径的实体
  deleteEntity(entity: Entity) {
    this.viewer.entities.remove(entity);
  }

  /**
   *创建对应轨迹的实体，或者是所有轨迹的实体，谨防重复创建
   * @param index 要创建实体的对应轨迹的索引，undefined表示创建所有实体
   */
  createEntities(index?: number, entityOption?: Entity.ConstructorOptions) {
    const create = (i: number) => {
      const twj = this.tracksWithJulidate[i];
      let property = this.getProperty(twj);
      let startTime = twj[0].time;
      let stopTime = twj[twj.length - 1].time;
      let modelUrl = this.modelUrls[i];
      const entity = this.createEntity({
        startTime,
        property,
        stopTime,
        modelUrl,
        options: entityOption,
      });
      this.entities.push(entity);
      // 记录轨迹和实体的映射
      this.trackEntityMap.set(twj, entity);
    };
    if (index !== undefined) {
      create(index);
      return;
    }
    // 创建实体
    for (let i = 0; i < this.tracksWithJulidate.length; i++) {
      create(i);
    }
  }

  /**
   * 创建实体的方法
   * @param option
   * @returns
   */
  createEntity(option: CreateEntityOption) {
    let { startTime, stopTime, property, modelUrl, options } = option;
    let modelOption: ModelGraphics.ConstructorOptions = {};
    if (modelUrl)
      modelOption = {
        uri: modelUrl,
        minimumPixelSize: 128,
        maximumScale: 2000,
      };
    return new Entity({
      availability: new TimeIntervalCollection([
        new TimeInterval({
          start: startTime,
          stop: stopTime,
        }),
      ]),
      // 采样属性，为一系列，点+时间组成的坐标属性，每两个坐标属性之间是匀速直线运动
      position: property,
      // 模型朝向，为速度的实时方向
      orientation: new VelocityOrientationProperty(property),
      model: modelOption,
      // 轨迹路径
      path: {
        resolution: 1,
        material: Color.YELLOW,
        width: 2,
      },
      ...options,
    });
  }
  // 获取采样属性
  getProperty(track: TrackWithJuliDate) {
    const property = new SampledPositionProperty();
    for (let { position, time } of track) {
      property.addSample(time, position);
    }
    return property;
  }

  // 获取一个路径点位时间
  static getSiteTime(trackCate3: Cartesian3[], speed = 25) {
    let timeSum = 0;
    let siteTimes = [0];

    for (let i = 1; i < trackCate3.length; i++) {
      let prePoint = trackCate3[i - 1];
      let curPoint = trackCate3[i];
      timeSum += TracksManager.getSpaceDistance(prePoint, curPoint) / speed;
      siteTimes.push(timeSum);
    }
    return { timeSum, siteTimes };
  }
  // 计算多个点的总距离
  static getSpaceDistance(...positions: Cartesian3[]): number {
    let sumDistance = 0;
    for (let i = 1; i < positions.length; i++) {
      let prePoint = Cartographic.fromCartesian(positions[i - 1]);
      let curPoint = Cartographic.fromCartesian(positions[i]);
      // 计算当前点到上一点的距离
      const geodesic = new EllipsoidGeodesic();
      geodesic.setEndPoints(prePoint, curPoint);
      const horizontalDistance = geodesic.surfaceDistance;
      const verticalDistance = prePoint.height - curPoint.height;
      sumDistance += Math.sqrt(
        Math.pow(horizontalDistance, 2) + Math.pow(verticalDistance, 2)
      );
    }
    return sumDistance;
  }

  toTracksWithJuliDate(tracks: Track[] | TrackWithTime[]): TrackWithJuliDate[] {
    // 如果已经是trackWithTime 直接返回
    if ((tracks as TrackWithTime[])[0][0].time) {
      //将时间转为julidate
      return this.getTracksWithJulidateFromTrackWithTime(
        tracks as TrackWithTime[]
      );
    }
    return this.getTracksWithJulidateFromTracks(tracks as Track[]);
  }
  getTracksWithJulidateFromTracks(tracks: Track[]) {
    const tracksWithJuliDate: TrackWithJuliDate[] = [];
    this.startTime = JulianDate.fromDate(new Date());
    let stopTime = this.startTime.clone();
    // 如果有时间，根据时间计算速度
    if (this.times) {
      this.speeds = [];
      this.times.forEach((time, i) => {
        let speed = TracksManager.getSpaceDistance(...tracks[i]) / time;
        this.speeds.push(speed);
      });
    }
    // 按照速度计算日期
    tracks.forEach((track, i) => {
      let { siteTimes, timeSum } = TracksManager.getSiteTime(
        track,
        this.speeds[i]
      );
      tracksWithJuliDate.push([]);
      siteTimes.forEach((time, j) => {
        tracksWithJuliDate[i].push({
          position: track[j],
          time: JulianDate.addSeconds(stopTime, time, new JulianDate()),
        });
      });
      stopTime = JulianDate.addSeconds(stopTime, timeSum, new JulianDate());
    });

    this.stopTime = stopTime;
    return tracksWithJuliDate;
  }
  getTWJFromTrack(
    track: Track,
    startTime: JulianDate,
    speed = 25,
    time?: number
  ) {
    const twj: TrackWithJuliDate = [];
    if (time) speed = TracksManager.getSpaceDistance(...track) / time;
    const { siteTimes } = TracksManager.getSiteTime(track, speed);
    track.forEach((carte3, i) => {
      twj.push({
        position: carte3,
        time: JulianDate.addSeconds(startTime, siteTimes[i], new JulianDate()),
      });
    });
    return twj;
  }
  getTWJFromTWT(track: TrackWithTime) {
    const twj: TrackWithJuliDate = [];
    track.forEach(({ position, time }) => {
      twj.push({ position, time: JulianDate.fromDate(time) });
    });
    return twj;
  }
  // 从TrackWithTime获取轨迹
  getTracksWithJulidateFromTrackWithTime(tracks: TrackWithTime[]) {
    const tracksWithJuliDate: TrackWithJuliDate[] = [];
    tracks.forEach((track) => {
      tracksWithJuliDate.push(this.getTWJFromTWT(track));
    });
    this.getStartAndStopFromTracksWithTime(tracks);
    return tracksWithJuliDate;
  }
  // 获取startTime和stopTime 从TrackWithTime
  getStartAndStopFromTracksWithTime(tracks: TrackWithTime[]) {
    let minStart = tracks[0][0].time;
    let maxEnd = tracks[0][tracks[0].length - 1].time;
    for (let i = 1; i < tracks.length; i++) {
      let start = tracks[i][0].time;
      let end = tracks[i][tracks[i].length - 1].time;
      if (minStart > start) minStart = start;
      if (maxEnd < end) maxEnd = end;
    }
    this.startTime = JulianDate.fromDate(minStart);
    this.stopTime = JulianDate.fromDate(maxEnd);
    return [minStart, maxEnd];
  }
  getStartAndStopFromTracksWithJulidate(tracks: TrackWithJuliDate[]) {
    if (tracks.length == 0) {
      this.startTime = new JulianDate();
      this.stopTime = new JulianDate();
    }
    let minStart = tracks[0][0].time;
    let maxEnd = tracks[0][tracks[0].length - 1].time;
    for (let i = 1; i < tracks.length; i++) {
      let start = tracks[i][0].time;
      let end = tracks[i][tracks[i].length - 1].time;
      if (JulianDate.compare(minStart, start) > 0) minStart = start;
      if (JulianDate.compare(maxEnd, end) < 0) maxEnd = end;
    }
    this.startTime = minStart;
    this.stopTime = maxEnd;
    return [minStart, maxEnd];
  }

  fillArray<T>(value: T, length?: number) {
    if (!length) return [];
    const arr = [];
    for (let i = 0; i < length; i++) {
      arr.push(value);
    }
    return arr;
  }
  /**
   * 传入成员类型可能为一个二维数组或一维数组，返回一个二维数组
   * @param arr
   */
  getTwoDimensionArray<T1, T2>(
    arr: T1 | Array<T1> | T2 | Array<T2>
  ): Array<T1> | Array<T2> {
    if (Array.isArray(arr) && Array.isArray(arr[0])) {
      //如果arr 是一个二维数组直接返回
      return arr;
    } else {
      //否则，将其转换为二维数组返回
      return [arr] as Array<T1> | Array<T2>;
    }
  }
  removeItemInArr<T>(arr: Array<T>, item: T): Array<T> {
    const newArr: T[] = [];
    for (let ele of arr) {
      if (item === arr) continue;
      newArr.push(ele);
    }
    return newArr;
  }
}
