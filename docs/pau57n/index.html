一、子组件优化 Memo

1.引例

子组件不使用父组件的数据，父组件数据更新，子组件会更新。甚至父组件没有使用到更新的数据也会更新。😵

import React from 'react';
import { useEffect } from 'react';
import { useState } from 'react';
import Count from './Count';
const City = () => {
  const [count, setCount] = useState(0);
  useEffect(() => {
    console.log('父组件更新了');
  });
  return (
    <div>
      <button onClick={() => setCount(count + 1)}>count</button>
      <Count />
    </div>
  );
};

export default City;

import React from 'react';
import { useEffect } from 'react';

const Count = () => {
  useEffect(() => {
    console.log('子组件更新了');
  });
  return <div>子组件</div>;
};

export default Count;

解决方案

使用memo解决，此时跟子组件不相干的基本数据类型更新后，子组件不会更新

import React, { useEffect, memo } from 'react';

const Count = memo(() => {
  useEffect(() => {
    console.log('子组件更新了');
  });
  return <div>子组件</div>;
});

export default Count;

使用memo后，给子组件直接传递引用类型如对象和方法，父组件（相关引用类型未更新）更新后，子组件不会更新。

猜测，页面更新时，数据存储在了内存里，其引用(地址)并没有变化

新的问题

当给子组件传递一个方法，主要是匿名函数，箭头函数时，memo会失效父组件更新时子组件会更新

 <Count person={person} addCount={() => setCount(count + 1)} />

<Count
  person={function () {
    person;
  }}
  />

猜测，父组件没有保存匿名函数的相关引用在栈中，子组件保存了其引用，每次父组件在更新后，就会向子组件重新传递一个新的匿名函数，因此子组件那里就会认为状态改变，更新页面

验证:思路：将父组件传递的方法存放到redux里，并且比较前后两个方法的引用

import React, { memo } from 'react';
import { useEffect } from 'react';
import { useDispatch, useSelector } from 'react-redux';
const Count = memo((props) => {
  let $dispatch = useDispatch();
  let fooArr = useSelector((state) => state.fooSlice.functionArr);
  useEffect(() => {
    console.log(fooArr);
    console.log('子组件更新了');
    if (fooArr[0] && fooArr[1]) {
      console.log(fooArr[0] === fooArr[1]);
    }
  });

  return (
    <div>
      子组件
      {/* <button onClick={() => addCount(0)}>点击父组件count归0</button> */}
      <button onClick={() => props.addCount()}>点击父组件count归0</button>
      <button
        onClick={() => {
          $dispatch({
            type: 'fooSlice/pushFoo',
            payload: props.addCount,
          });
        }}
      >
        派发函数
      </button>
    </div>
  );
});

export default Count;

import React from 'react';
import { useEffect } from 'react';
import { useState, useMemo } from 'react';
import Count from './Count';
const City = () => {
  const [count, setCount] = useState(0);
  const [person, setPerson] = useState(() => {
    return { name: '张三', age: 18 };
  });
  useEffect(() => {
    console.log('父组件更新了');
  });
  const [flag, setFlag] = useState(true);
  const CountMemo = useMemo(() => {
    return <Count flag={flag} addCount={() => setCount(count + 1)} />;
  }, [flag]);
  return (
    <div>
      <button onClick={() => setCount(count + 1)}>{count}</button>
      <button onClick={() => setFlag(!flag)}>更新子组件</button>
      {/* 未更新版 */}
      {/* <Count person={person} addCount={setCount} /> */}

      {/* 更新版 */}
      {/* <Count person={person} addCount={() => setCount(count + 1)} /> */}
      {/* <Count
        person={function () {
          person;
        }}
      /> */}
      {/*useMemo版*/}
      {CountMemo}
    </div>
  );
};

export default City;

useMemo

针对以上问题， 解决方案，使用**useMemo**，语法：**useMemo(callback,depsarr)****callback function **返回一个值，可以是组件或者js相关的值，**depsarr array ** 数组，只有其中的值发生改变才会执行callback

import React, { useMemo } from 'react';
import { useState } from 'react';

const City = () => {
  const [city, setCity] = useState('武汉');
  const [count, setCount] = useState(0);
  const double = useMemo(() => {
    // 只有当依赖改变才会重新计算
    console.log('useMemo');
    return count * 2;
  }, [count]);
  return (
    <div>
      <p>
        <button
          onClick={() => {
            setCity(city + 1);
          }}
        >
          {city}
        </button>
      </p>
      <p>
        <button
          onClick={() => {
            setCount(count + 1);
          }}
        >
          {count}
        </button>
      </p>
    </div>
  );
};

export default City;
