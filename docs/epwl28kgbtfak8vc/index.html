  url: `https://js.arcgis.com/3.35/init.js`,
  css: "https://js.arcgis.com/3.35/esri/css/esri.css",

初始化地图，并添加天地图作为底图，把token换成你自己的天地图token即可

<template>
  <div id="map">
  </div>
</template>

<script setup lang="ts">
import { loadModules } from 'esri-loader'
import { onMounted } from 'vue';

const initMap = (mapId:string,tk:string ='b45990d3c06e4525d409c27a2133508a') => {
  return loadModules([
    "esri/map",
    "esri/layers/WebTiledLayer",
    "esri/SpatialReference",
  ]).then(([Map, WebTiledLayer, SpatialReference]) => {
    const esriMap = new Map(mapId, {
      basemap: "",
      zoom: 8,
      center: [120, 25],
      spatialReference: new SpatialReference({ wkid: 4490 }),
    });
    let url = 'http://{subDomain}.tianditu.com/img_w/wmts?layer=img&style=default&tilematrixset=w&Service=WMTS&Request=GetTile&Version=1.0.0&Format=&TileMatrix={level}&TileCol={col}&TileRow={row}&tk='+tk
    const tileLayer = new WebTiledLayer(url, {
      subDomains: ["t0", "t1", "t2", "t3", "t4", "t5", "t6", "t7"],
    });
    esriMap.addLayer(tileLayer);
    map=esriMap
  });
}
onMounted(()=>{
  initMap('map')
})
</script>

<style scoped>
#map{
  width:100%;
  height: 100vh;
}

封装一个基础天地图为底图的地图组件

由于模块的异步加载，会导致父组件在onMounted回调里不能拿到子组件暴露的map对象，因此给子组件添加一个load事件，用于获得map

<template>
  <div ref="mapRef" :style="style" :class="mapClass">
    <slot />
  </div>
</template>

<script setup lang="ts">

import { onMounted, ref, StyleValue } from 'vue';
import {initMap } from '@/common/ts/initMap'

const mapRef = ref()
const emit=defineEmits(['load'])
const map=ref()
 onMounted(async () => {
  map.value=await initMap(mapRef.value)
  emit('load',map.value)
})
defineProps<{
  style?:StyleValue,
  mapClass?:string,
}>()
defineExpose({
  map: map
})
</script>

<style scoped></style>

import { loadModules } from 'esri-loader'

export const initMap = (mapId:string|HTMLElement,tk ='b45990d3c06e4525d409c27a2133508a',) => {
  return loadModules([// 注意异步操作
    "esri/map",
    "esri/layers/WebTiledLayer",
    "esri/SpatialReference",
  ]).then(([Map, WebTiledLayer, SpatialReference]) => {
    const esriMap = new Map(mapId, {
      basemap: "",
      zoom: 8,
      center: [120, 25],
      spatialReference: new SpatialReference({ wkid: 4490 }),
      logo:false
    });
    let url = 'http://{subDomain}.tianditu.com/img_w/wmts?layer=img&style=default&tilematrixset=w&Service=WMTS&Request=GetTile&Version=1.0.0&Format=&TileMatrix={level}&TileCol={col}&TileRow={row}&tk='+tk
    const tileLayer = new WebTiledLayer(url, {
      subDomains: ["t0", "t1", "t2", "t3", "t4", "t5", "t6", "t7"],
    });
    esriMap.addLayer(tileLayer);
    return esriMap
  });
}
