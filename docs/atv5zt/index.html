function unique(...arr) {
  let len = arr.length;
  if (len <= 1) return arr;
  const myIndexOf = (arr, item) => {
    for (let i = 0; i < arr.length; i++) {
      if (arr[i] === item) return i;
    }
    return -1;
  };
  for (let i = 0; i < len; i++) {
    let index = myIndexOf(arr, arr[i]);
    if (index != -1 && index < i) arr[i] = undefined;
  }
  let start = 1;
  let j = 2;
  while (start < len && j < len) {
    if (arr[start] !== undefined) {
      start++;
      j = start + 1;
      continue;
    }
    while (j < len) {
      if (arr[j] !== undefined) {
        arr[start] = arr[j];
        arr[j] = undefined;
        start++;
        j++;
        break;
      }
      j++;
    }
  }
  for (let i = 1; i < len; i++) {
    if (arr[i] == undefined) {
      arr.length = i;
      break;
    }
  }
  return arr;
}
