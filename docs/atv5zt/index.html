实现快速排序

基本思想是:通过一趟排序将要排序的数据分割成独立的两部分，其中一部分的所有数据都比另外一部分的所有数据都要小，然后再按此方法对这两部分数据分别进行快速排序，整个排序过程可以递归进行，以此达到整个数据变成有序序列。

核心思想

  1. 任取一个元素为基准
  2. 将待排序的元素进行分块，比基准大的元���移动到基准元素的右侧，比基准小的移动到左侧，从而一趟排序过程就可以锁定基准元素的最终位置
  3. 对以上两块进行上述规则的排序
  4. 将左边排序好的和基准元素和右边排序好的拼接起来。递归所在
function quickSort(nums){
  let len=nums.length
  if(len<=1) return nums
  let base=nums[0]
  let left=[],right=[]
  for(let i=1;i<len;i++){
    if(base>nums[i]) left.push(nums[i])
    else right.push(nums[i])
  }
  return [...quickSort(left),base,...quick(right)]
}

function defineReactive(target,key,value){
	if(typeof value==='object') observe(value)
  Object.defineProperty(target,key,{
    get(){
      //执行获取前的操作
      return value
    }
    set(val){
      value=val
      // 渲染视图
    }
  })
}
function observe(target){
  if(target&&typeof target!='object') return target
  for(let key in target){
    defineReactive(target,key,target[key])
  }
}

实现dom树转js对象

思路： :::info dom对象中记录哪些内容，当前节点的类型type=TagName，子节点children :::

function domToObject(root) {
  // 如果不是元素节点，就直接返回
  if (root.nodeType != '1') return null
  let node = {}
  node.type = root.tagName
  node.attr = {}
  for (let i = 0; i < root.attributes.length; i++) {
    let attr = root.attributes[i]
    node.attr[attr.name] = attr.value
  }
  if (root.children) {
    node.children = []
    for (let n of root.children) {
      node.children.push(domToObject(n))
    }
  }
  return node
}

实现instanceof

思路： :::warning 检查类型是否出现在对象的原型链上，如果出现，就返回true，否则返回false。如何获得原型链。通过__proto__获得 :::

function myInstanceof(obj,origin){
  if(obj===null||obj===undefined) return false
  let proto=obj.__proto__
  while(proto){
    if(proto===origin.prototype) return true
    proto=proto.__proto__
  }
  return false
}

数组去重

思路： :::warning

  1. set集合
  2. map对象
  3. 不使用任何数据结构，和数组的方法。并且不新建数组。也就是说在原数组中进行改变。思路有如下两步
    1. 先在数组中将重复的元素设置为undefined
    2. 将数组中的元素向前移动，并修改长度 :::
function unique(arr){
  return Array.from(new Set(arr))
}

function unique2(arr) {
  let map = new Map();
  arr.forEach((item) => map.set(item, item));
  return Array.from(map.keys());
}

function unique(...arr) {
  let len = arr.length;
  if (len <= 1) return arr;
  const myIndexOf = (arr, item) => {
    for (let i = 0; i < arr.length; i++) {
      if (arr[i] === item) return i;
    }
    return -1;
  };
  for (let i = 0; i < len; i++) {
    let index = myIndexOf(arr, arr[i]);
    if (index != -1 && index < i) arr[i] = undefined;
  }
  let start = 1;
  let j = 2;
  while (start < len && j < len) {
    if (arr[start] !== undefined) {
      start++;
      j = start + 1;
      continue;
    }
    while (j < len) {
      if (arr[j] !== undefined) {
        arr[start] = arr[j];
        arr[j] = undefined;
        start++;
        j++;
        break;
      }
      j++;
    }
  }
  for (let i = 1; i < len; i++) {
    if (arr[i] == undefined) {
      arr.length = i;
      break;
    }
  }
  return arr;
}

function flat(arr){
  for(let i=0;i<arr.length;i++){
    while(arr[i] instanceof Array){
      arr=[...arr.slice(0,i),...arr[i],...arr.slice(i+1)]
    }
  }
  return arr
}

function deepCopy(target){
  if(typeof target!='object'||target===null) return target
  let res={}
  if(Object.prototype.toString.call(target)==='[object Array]')
    res=[]
  for(let key in target) res[key]=deepCopy(target[key])
  return res
}

实现reduce

思路： :::info reduce((pre,cur,i,arr)=>{},initialValue)

  1. pre表示上一个回调的结果
  2. cur表式当前遍历的元素
  3. i当前的索引
  4. initialValue，pre初始值，如果没有，pre默认为数组的第一个元素，此时cur从第二个元素开始 :::
Array.prototype.myReduce=function(fn,initVal){
  const arr=this
  let start=1
  let pre=arr[0]
  if(initVal!==undefined){
    start=0
    pre=initVal
  } 
  for(let i=start;i<arr.length;i++){
    let cur=arr[i]
    pre=fn(pre,cur,i,arr)
  }
  return pre
}

解决精度丢失问题

思路：

转换位整数后进行计算，再转换为小数

function add(num1,num2){
  if(num1%1==0&&num2%1==0) return num1+num2
  let decimal1=num1.toString().split('.')[1]||''
  let decimal2=num2.toString().split('.')[1]||''
  let e=10**Math.max(decimal1.length,decimal2.length)
  return (num1*e+num2*e)/e
}

函数节流

// 节流，第一次立即触发，此后每次间隔指定时间触发
function throttle(func, wait = 500) {
	
  let previous = 0;
  return function () {
    let now = +new Date();
    let context = this;
    if (now - previous >= wait) {
      func.apply(context, arguments);
      previous = now; // 执行后更新 previous 值
    }
  }
}

函数防抖

// 思路 延迟用户执行的代码，在一定范围时间内，重复触发，取消上次用户执行的代码，执行最新的一次。
function debounce(func,delay=500){
  let timer=null
  return function(){
    let self=this
    if(timer) {
      clearTimeout(timer)
    }
    setTimeout(()=>{
      func.apply(self,arguments)
      timer=null
    },delay)
  }
}
