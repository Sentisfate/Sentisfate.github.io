能否连接形成数组

2022-09-22 02:16

题目描述

解法1 DFS+剪枝

思路：

遍历arr，如果当前元素在pieces[i]中的第一个元素的话，就比较pieces[i]中的每个元素，如果与arr中的元素相等，就从arr比较完之后的第一个元素开始，继续重复上述过程比较。如果不相等，由于题目条件为不重复，可以进行剪枝直接返回false

临界条件： :::info

::: 证明： :::tips

:::

/**
 * @param {number[]} arr
 * @param {number[][]} pieces
 * @return {boolean}
 */
var canFormArray = function(arr, pieces) {
    // 遍历arr，如果当前元素在pieces[i]中的第一个元素出现，就比较pieces[i]中的每个元素，否则，换到pieces[i+1]
    // 深度优先搜索，终止条件为，arr中开始的位置为最后arr的长度时，返回true，当pieces遍历到最后一个元素时，返回false

    const dfs=(arr,start,pieces)=>{
        if(start==arr.length) return true

        for(let i=0;i<pieces.length;i++){
                let len=pieces[i].length
            if(arr[start]==pieces[i][0]){
                // 比较
                for(let j=1;j<len;j++){
                    if(arr[start+j]!=pieces[i][j]) return false //因为不会重复，所以，不等就无解
                }
                //相等，就递归比较下一个部分
                return dfs(arr,start+len,pieces)
            }
        }
        return false
    }
    return dfs(arr,0,pieces)
};
