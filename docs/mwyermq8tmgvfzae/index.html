裁剪空间是webgl能绘制的范围，-1~1，0在中间，当坐标超出这个范围时，不会显示出范围外的部分，此外，x和y方向的单位长度为canvas的宽高的一半，互相不一定相等

function toClipSpace(positions,width,height,centerx,centery){
  for(let i=0;i<positions.length;i+=2){
    let x=positions[i]
    let y=positions[i+1]
    positions[i]=2/width*(x-centerx)
    positions[i+1]=-2/height*(y-centery)
  }
}

以及glsl语言里矢量的特性，矢量转置后可以看作是相等的。

  // 齐次变换矩阵
  // vec3 result=position*m; 矩阵写错后意外发现右乘能实现效果😂
  vec3 result=m*position;
  gl_Position = vec4(result.x,result.y,0,1);
}

function getMat3(canvas) {
  const rec = canvas.getBoundingClientRect();
  const center = {
    x: (rec.left + rec.right) / 2,
    y: (rec.top + rec.bottom) / 2,
  };
  const { x, y } = center;
  const w = canvas.width;
  const h = canvas.height;
  return [
    2 / w,
    0.0,
    0,
    0.0,
    -2 / h,
    0,
    (-2.0 * x) / w,
    (2.0 * y) / h,
    1.0,
  ];
}

变换矩阵

因为有平移变换，需要引入齐次坐标。

不引入齐次坐标的变换矩阵，只能表达相对于原来原点的变换，不能表达平移，因此需要多加一个维度来表示。

注意glsl里矩阵列优先，也就是说写成下面的格式

mat3 m=mat3(sx,0.0,0,0.0,sy,0.,tx,ty,1.0);//sx为x缩放倍数，tx为位移量
