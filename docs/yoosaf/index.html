统计子串中的唯一字符

2022-09-06 03:14

题目描述

解法1. 穷举法

思路：

采用滑动窗口来寻找所有的子串，并统计子串的唯一字符的个数

临界条件： :::info 窗口的起始位置索引 **i**，窗口的宽度 **width**,字符串的长度**length**

:::

/**
 * @param {string} s
 * @return {number}
 */
var uniqueLetterString = function(s) {
    //获得一个字符串的无重复字符的数量
    const countUniqueChars=(s)=>{
        let count=0
        let map=new Map()
        for(let i=0;i<s.length;i++){
            let c=map.get(s[i])||0
            map.set(s[i],c+1)
        }
        for(let val of map.values()){
            if(val==1){
                count++
            }
        }
        return count
    }
    let sum=0
    let length=s.length
    let width=length
    // 以一个滑动窗口来获取子串
    while(width>=1){
        for(i=0;i+width<=length;i++){
            let subStr=s.slice(i,width+i)
            console.log(subStr)
            sum+=countUniqueChars(s.slice(i,width+i))
        }
        width--
    }
    console.log(sum)
    return sum
};
