方法一、document.execCommand

思路，使用input来实现选中文本(这里使用textarea)，再通过document.execCommand来进行复制。

/**
 *
 * @param {string} text 要复制的文本内容
 * @returns 返回一个boolean值，为true则复制成功，false则失败
 */
export function copy(text) {
  const textarea = document.createElement('textarea');
  textarea.setAttribute('readonly', 'readonly'); //设置只读属性防止手机弹出软键盘
  textarea.value = text;
  document.body.appendChild(textarea);
  textarea.select();
  let res=document.execCommand('copy');
  textarea.remove()
  return res
}

方法二、Clipboard

系统剪贴板暴露于全局属性 Navigator.clipboard 之中。浏览器的适配情况

clipboard提供了操作剪贴板的方法。

read

从剪贴板读取数据（比如图片），返回一个 Promise 对象。其会兑现一个ClipboardItem数组。 The promise is rejected if permission to access the clipboard is not granted.权限被禁止promise会被拒绝

read方法可以获取任意类型数据。数据通过ClipboardItem获得。

ClipboardItem

属性

  • types 只读，一个文件类型数组，如'text/plain'``"text/html"
    • 更多MIME类型参看 https://www.iana.org/assignments/media-types/media-types.xhtml

方法

  • getType() 返回一个能兑现按照指定类型blob对象的promise

对于文本内容，可以通过Blob.prototype.text()读取，注意返回的是promise。

<script>
  function foo() {
    navigator.permissions
      .query({ name: "clipboard-read" })
      .then((result) => {
        // If permission to read the clipboard is granted or if the user will
        // be prompted to allow it, we proceed.
        console.log(result)
        if (result.state == "granted" || result.state == "prompt") {
          navigator.clipboard.read().then((data) => {
            console.log(data)
            for (let item of data) {
              // console.log(item.getType(''))
              for (const type of item.types) {
                item.getType(type).then(blob => {
                  console.log(blob)
                  if (type === 'text/plain' || type === 'text/html') {
                    blob.text().then(text => console.log(text))
                  }
                })
              }
            }

          });
        }
      });
  }
</script>

由于promise过多，可以通过async，await简化，下面是一个初步的封装。

async function getContentFromClipboard() {
  let res = []
  const clipboardItems = await navigator.clipboard.read()
  for (const item of clipboardItems) {
    const map = new Map()
    for (const type of item.types) {
      const blob = await item.getType(type)
      if (type === 'text/plain' || type === 'text/html') {
        map.set(type, await blob.text())
        continue
      }
      map.set(type, blob)
    }
    res.push(map)
  }
  return res
}

write

将数据写到剪贴板数据被写入时，resolve回调执行，

async function copy(blob) {
  try {
    await navigator.clipboard.write([
      new ClipboardItem({
        [blob.type]: blob
      })
    ])
    console.log('copy success')
  } catch (err) {
    console.log(err.name, err.message)
  }
}

writeText

复制文本

async function copyText(text) {
  try {
    await navigator.clipboard.writeText(text)
    console.log('copy success')
  } catch (err) {
    console.log(err.name, err.message)
  }
}

总结

综合上面两种方法，实现一个通用复制文本的功能。

function copyByExec(text) {
  const textarea = document.createElement('textarea');
  textarea.setAttribute('readonly', 'readonly'); //设置只读属性防止手机弹出软键盘
  textarea.value = text;
  document.body.appendChild(textarea);
  textarea.select();
  let res = document.execCommand('copy');
  textarea.remove()
  return res
}
export async function copyText(text) {
  // 如果浏览器不支持clipboard
  if (!navigator.clipboard) {
    return copyByExec(text)
  }
  try {
    await navigator.clipboard.writeText(text)
    return true
  } catch (err) {
    return copyByExec(text)
  }
}
