单词拆分

2022-09-07 04:42

题目描述

解法1 dfs

思路：

一个单词如果要能拆分，必定这个字符串的第一个字符开始到某一字符结束所构成的单词能够在字典种找到。并且剩下的字符串可以按照同样的规律来找。很显然，可以通过递归的方式来查找

临界条件： :::info 当递归函数要查询的字符串为空时，说明可以拆分。返回true ::: 

/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function(s, wordDict) {
    if(wordDict.includes(s)){
        return true
    }  
    // 如果可以由字典种组成的话，则字符串满足从第一个字符开始到某一位置的单词能在字典种找到
    // 拿么剩下的字符串按照同样的条件一样可以在字典种找到对应的单词
    // 因此可以考虑使用递归的方式来查找
    const dfs=(s,wordDict)=>{
        //当递归的字符串长度为1时，说明前面有一种拆分方式可以成功
        if(s.length==0){
            return true
        }
        for(let i=0;i<s.length;i++){
            let word=s.slice(0,i+1)
            console.log(word)
            //如果当前的单词可以在字典种找到的话，就递归搜索后面的字符串
            if(wordDict.includes(word)){
                if(dfs(s.slice(i+1),wordDict)){
                    return true
                }
            }
        }
        return false
        //没有一个能够成立的话就
    }
    return dfs(s,wordDict)
};

上面的方法时间复杂度会比较高，遇到某些情况可能会超时，如 :::tips “aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaab” [“a”,“aa”,“aaa”,“aaaa”,“aaaaa”,“aaaaaa”,“aaaaaaa”,“aaaaaaaa”,“aaaaaaaaa”,“aaaaaaaaaa”] :::

因此可以用一个数组记忆某一节点后面的递归结果即可

/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function (s, wordDict) {
  if (wordDict.includes(s)) {
    return true;
  }
  // 如果可以由字典种组成的话，则字符串满足从第一个字符开始到某一位置的单词能在字典种找到
  // 拿么剩下的字符串按照同样的条件一样可以在字典种找到对应的单词
  // 因此可以考虑使用递归的方式来查找

  // 用于记录从当前字符串第i个开始的字符串能否拆分成单词
  let memo = new Array(s.length);
  //start 记录的是上一次字符的位置
  const dfs = (s, wordDict, start) => {
    //当递归的字符串长度为1时，说明前面有一种拆分方式可以成功
    if (s.length == 0) {
      return true;
    }
    if (memo[start] != undefined) {
      return memo[start];
    }
    //假定当前开始位置不能被拆分
    memo[start] = false;
    for (let i = 0; i < s.length; i++) {
      //如果从当前位置开始的字符串能够拆分的话就返回
      //如果当前位置开始的字符串的状态未知的话，就进行递归确认当前的
      let word = s.slice(0, i + 1);
      //如果当前的单词可以在字典种找到的话，就递归搜索后面的字符串
      if (wordDict.includes(word)) {
        //递归标记标记从该字符后面开始的字符串可以被拆分,如果后面的字符可以拆分，当前开始的位置就可以拆分，否则当前开始的位置的字符串就不能拆分
        if (dfs(s.slice(i + 1), wordDict, start + i + 1)) {
          memo[start] = true;
          break;
        }
      }
    }
    return memo[start]
    //没有一个能够成立的话就
  };
  return dfs(s, wordDict, 0);
};

解法2 动态规划法

:::tips 思路：用一个数组dp来记录前i个字符组成的字符串是否可以被拆分，并记录下i的索引。前j (j>i)个字符串是否可以被拆分成单词就可以转换为

  • i+1 ~ j个字符组成的字符串是否在字典中
  • dp中记录的i+1~j是否在字典中 :::
/**
 * @param {string} s
 * @param {string[]} wordDict
 * @return {boolean}
 */
var wordBreak = function(s, wordDict) {
    //考虑用动态规划来解决
    // 用一个数组来存储前i为字符是否能够被拆分成单词,并记下这个i
    let n=s.length
    // 默认存储前0个
    let dp=[-1]
    let d=1
    for(let i=0;i<n;i++){
        for(let j=dp.length-1;j>=0;j--){
            if(wordDict.includes(s.slice(dp[j]+1,i+1))){
                dp[d++]=i
                break
            }
        }
    }
    return dp[dp.length-1]==n-1
};
