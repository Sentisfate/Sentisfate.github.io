基本思想

入门 使用redux

  1. 安装依赖

**npm i redux****npm i react-redux**

  1. 创建store/index.js
import { createStore } from 'redux';
const initState = {
  count: 1,
};

// reducer -->pinia 中的actions
function reducer(state = initState, action) {
  switch (action.type) {
    case 'counter/incremented':
      return { count: state.count + 1 };
    case 'counter/decremented':
      return { count: state.count - 1 };
    default:
      return state;
  }
}

let store = createStore(reducer);
export default store;
  1. 配置main.jsx
import React from 'react';
import ReactDOM from 'react-dom/client';
import App from './App';
import store from './store';
import { Provider } from 'react-redux';
ReactDOM.createRoot(document.getElementById('root')).render(
  <Provider store={store}>
    <App />
  </Provider>
);
  1. 使用**store**中的数据 **store.getState().状态名**
import React, { useEffect } from 'react';
import { useStore } from 'react-redux';
const City = () => {
  const store = useStore();
  useEffect(() => {
    console.log(store);
  }, []);
  return <div>{store.getState().count}</div>;
};

export default City;

使用reduxjs/toolkit

  1. 安装依赖

**npm install @reduxjs/toolkit react-redux**

  1. 使用**createSlice**定义切片，一个包含了state和action的逻辑块
import { createSlice } from '@reduxjs/toolkit';
export const counterSlice = createSlice({
  name: 'counter',
  initialState: {
    value: 0,
  },
  reducers: {
    increment: (state) => {
      state.value += 1;
    },
  },
});

import { createSlice } from '@reduxjs/toolkit';
export const CitySlice = createSlice({
  name:'city',
  initialState: {
    value: '武汉',
  },
  reducers: {
    //注意这里的第二个参数是dispatch({})派发过来的对象，这里将其解构只用了payload
    changeCity(state,{payload}) {
      state.value = payload;
    },
  },
});
  1. 集中管理切片，注意易错点，**slice**里用的是**reducers**，而**store**里用的是**reducer**

💡注意每个 **reducer** 只负责管理全局 state中它负责的一部分。每个 **reducer** 的 state 参数都不同，分别对应它管理的那部分 state 数据。全局总的state也就是**store**的**state**只有一个**reducer**负责，这个reducer由每个切片的reducer组成，因此configStore里的配置选项不写reducers，而每个切片有许许多多个状态，这些状态都有对应的reducer负责，一般肯定不止一个，所以切片里的选项为**reducers**

import { configureStore } from '@reduxjs/toolkit';
import { counterSlice } from './modules/count';
import { CitySlice } from './modules/city';
export default configureStore({
  // reducer: counterSlice.reducer,
  //只有一个切片可以像上面一样写，但是基本上不会只有一个
  reducer: {
    city: CitySlice.reducer,
    counter: counterSlice.reducer,
  },
});
  1. main.jsx中使用Provider引入store
import React from 'react';
import ReactDOM from 'react-dom/client';
import App from './App';
import store from './store';
import { Provider } from 'react-redux';
ReactDOM.createRoot(document.getElementById('root')).render(
  <Provider store={store}>
    <App />
  </Provider>
);
  1. 在组件中使用redux的数据
    1. 获取**state**** **state为一个对象，每个属性是一个slice的状态，属性名为slice 的**name**，因此**name**要唯一
import { useSelector } from 'react-redux';

...

  const state = useSelector((state) => {
    console.log(state);
    return state;
  });

  1. 获取对应属性的值可以这样写
let city=useSelector(state=>state.city.value)
let count=useSelector(state=>state.counter.value)
  1. 使用**useDispatch**向**redux**派发事件，改变状态

语法: **$dispatch({type:'slicename/reducer',payload:value...})**** 注意：**{type:'slicename/reducer',key:value}** 会作为reducer的第二个参数,可以自定义键值对来传递数据.**

import { useSelector, useDispatch } from 'react-redux';
...
  let $dispatch = useDispatch();
  const increment = () => {
    $dispatch({
      type: 'counter/increment',
    });
  };
  return (
    <div>
      <button onClick={increment}>{count}</button>
    </div>
  );

...
  const [newCity, setNewCity] = useState('');
  const changeCity = () => {
    $dispatch({
      type: 'city/changeCity',
      payload: newCity,
    });
  };
...
      <p>
        <input
          type="text"
          value={newCity}
          onChange={(e) => {
            setNewCity(e.target.value);
          }}
        />
        <button onClick={changeCity}>{city}</button>
      </p>

:::info 派发给store的事件，会根据type来确定由具体哪个slice的哪个reducer来执行，再根据payload传递来的数据替换原有的状态。 :::

总结

了解上述流程，再来看这张图，就基本明白了当事件触发时，引起了向store的派发操作，根据派发的类型和负载，相应的reducer就会操作状态，从而引发视图的改变。
