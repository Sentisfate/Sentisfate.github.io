function createQueue() {
  const list = [];
  let offset = 0;

  const dequeue = () => {
    let res = list[offset];
    list[offset] = undefined;
    offset++;

      // åˆ é™¤å‰é¢çš„ç©ºæ•°æ®
      list.splice(0, offset);
      offset = 0;
    }
    return res;
  };
  const enqueue = (...arr) => {
    list.push(...arr);
  };
  const pop = () => {
    return list.pop();
  };

  const obj = {
    dequeue,
    enqueue,
    pop,
  };

    },
    set(target, prop, value) {
      let index = Number(prop);
      if (isNaN(index)) return;
      target[index + offset] = value;
    },
  });
  return p;
}
