var buildTree = function (preorder, inorder) {
  const helper = (preorder, inorder) => {
    //中序遍历为空时，说明此节点为空,没有左右子树
    if (inorder.length == 0) {
      return null;
    }
    let rootVal = preorder.shift();

    let root = new TreeNode(rootVal);
    //找到根节点的左右子树,中序遍历中根节点左边的是左子树的中序遍历结果，右边同理
    let index = inorder.indexOf(root.val);
    // console.log(index);
    let leftInorder = inorder.slice(0, index);
    let rightInorder = inorder.slice(index + 1);

    //左节点为左子树的根节点，右节点为右子树的根节点

    root.left = helper(preorder, leftInorder);
    // 左子树遍历完
    root.right = helper(preorder, rightInorder);
    return root;
  };
  return helper(preorder, inorder);
};

var buildTree = function (preorder, inorder) {
  if (inorder.length == 0) {
    return null;
  }
  //中序遍历为空时，说明此节点为空,没有左右子树
  //指向当前的树的根节点
  let rootVal = preorder[0];
  const root = new TreeNode(rootVal);
  let cur = root;
  let stack = [];
  for (let i = 1, j = 0; i < preorder.length; i++) {
    // 如果当前根节点的左子树不为空，则前序遍历的下一个值就是左节点值
    if (cur.val != inorder[j]) {
      cur.left = new TreeNode(preorder[i]);
      stack.push(cur);
      cur = cur.left;
    } else {
      // 中序遍历的指针移向下一个节点
      j++;
      // 当前前序遍历的节点，为当前子树中序遍历的第一个节点，说明没有左子树为空，前序遍历的下一个节点就是当前根节点的右节点或者某个祖先节点的右节点
      //
      while (stack.length > 0 && stack[stack.length - 1].val == inorder[j]) {
        cur = stack.pop();
        j++;
      }
      // console.log(cur.val);
      cur.right = new TreeNode(preorder[i]);
      cur = cur.right;
    }
  }
  return root;
};

function TreeNode(val, left, right) {
  this.val = val === undefined ? 0 : val;
  this.left = left === undefined ? null : left;
  this.right = right === undefined ? null : right;
}
/**
 * @param {number[]} inorder
 * @param {number[]} postorder
 * @return {TreeNode}
 */
var buildTree = function (inorder, postorder) {
  const helper = (inorder, postorder) => {
    if (inorder.length == 0) {
      return null;
    }
    let rootVal = postorder.pop();
    let index = inorder.indexOf(rootVal);
    const root = new TreeNode(rootVal);
    let leftInorder = inorder.slice(0, index);
    let rightInorder = inorder.slice(index + 1);

    root.right = helper(rightInorder, postorder);
    root.left = helper(leftInorder, postorder);

    return root;
  };
  return helper(inorder, postorder);
};

function TreeNode(val, left, right) {
  this.val = val === undefined ? 0 : val;
  this.left = left === undefined ? null : left;
  this.right = right === undefined ? null : right;
}
/**
 * @param {number[]} inorder
 * @param {number[]} postorder
 * @return {TreeNode}
 */
var buildTree = function (inorder, postorder) {
  if (inorder.length == 0) {
    return null;
  }
  let last = postorder.length - 1;
  const root = new TreeNode(postorder[last]);
  //后序遍历可以看成是根右左的倒序，从后往前遍历，可以一次获得右左节点。
  // 对于后序遍历中相邻的两个节点m，n，m节点可能是n节点的右节点，或者是祖先节点的左节点
  const s = [];
  let j = last;
  let cur = root; //n
  // postorder[i]//m
  for (let i = last - 1; i >= 0; i--) {
    //
    //什么时候m是右节点,也就是当前的根节点是否有右子树
    //对于中序遍历而言，其最后一个节点，可能为根节点，或者为根节点的右节点
    // 也就是说，如果该结点不等于根节点，就是根节点的右节点，
    if (cur.val != inorder[j]) {
      cur.right = new TreeNode(postorder[i]);
      s.push(cur);
      cur = cur.right;
      continue;
    }
    // 相等的时候，说明以当前节点cur为根节点的子树没有右节点
    //判断postorder[i]是否为cur左节点，就只要判断中序遍历当前的节点和栈顶的元素是否相等即可
    j--;
    while (s.length > 0 && s[s.length - 1].val == inorder[j]) {
      cur = s.pop();
      j--;
    }
    //postorder[i]为cur的左节点
    cur.left = new TreeNode(postorder[i]);
    cur = cur.left;
  }
  return root;
};
