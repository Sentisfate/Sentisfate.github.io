经纬度这两个参数外，还有另外两个参数用来确定一个大圆的方程，

表示轨道与赤道的夹角的正切，也就是斜率，面面夹角相关知识，高三数学选修里应该介绍过，

表示大圆的朝向，这两个参数确定后则能确定一个大圆的位置。

这里在赤道上标注只是更方便直观展示朝向的影响，本质上朝向的计算应该是圆面的法向量在赤道面的投影与正东方向的夹角，至少我是这么理解的。

上面的方程已经能表示绝大部分的大圆，但是还有一类特殊的大圆极轨，无法表示，需要另加实现。

下面就是这样一个大圆的轨迹如何生成的代码实现了。

import { JulianDate } from "cesium"
import { Cartesian3 } from "cesium"
/**
 * 生成绕地轨道大圆轨迹
 * @param a 大圆轨道倾角
 * @param y 大圆轨道朝向
 * @param height 大圆上的点具地面的高度
 */
export function getTrack(a: number, y: number, height = 0) {
  const degreeUnit = Math.PI / 180
  const radianUnit = 180 / Math.PI
  if (a <= -90 || a >= 90) {
    alert('圆面倾角不能为90')
    return
  }
  const startTime = JulianDate.fromDate(new Date())
  // 定义轨道与地球赤道的倾角

  // 90为极轨
  let startLon = 0
  a = a * degreeUnit
  y = y * degreeUnit
  let tanA = Math.tan(a)
  console.log(tanA)
  const track = []
  // 以经度作为微分轴，将整个轨迹分为360份
  for (let i = startLon; i <= 360 + startLon; i++) {
    // 计算每一经度对应的经纬度
    const lon = i
    const lat = Math.atan(tanA * Math.sin(degreeUnit * lon + y)) * radianUnit
    track.push({
      position: Cartesian3.fromDegrees(lon, lat, height),
      time: JulianDate.addSeconds(startTime, i, new JulianDate()),
    });
  }
  return track;
}

function getPolarTrack(lon: number, height = 0, startTime = JulianDate.fromDate(new Date())) {
  const tracks = []
  for (let i = 0; i <= 360; i++) {
    tracks.push({
      position: Cartesian3.fromDegrees(lon, i, height),
      time: JulianDate.addSeconds(startTime, i, new JulianDate())
    })
  }
  return tracks
}

import { JulianDate } from "cesium"
import { Cartesian3 } from "cesium"
/**
 * 生成绕地轨道大圆轨迹
 * @param a 大圆轨道倾角
 * @param y 大圆轨道朝向
 * @param height 大圆上的点具地面的高度
 */
export function getBigCircleTrack(a: number, y: number, height = 0,startLon=0,startTime=JulianDate.fromDate(new Date())) {
  const degreeUnit = Math.PI / 180
  const radianUnit = 180 / Math.PI
  if (a == -90 || a == 90) {
    return getPolarTrack(startLon,height,startTime)
  }
  // 转为弧度
  a = a * degreeUnit
  y = y * degreeUnit
  let tanA = Math.tan(a)
  const track = []
  // 以经度作为微分轴，将整个轨迹分为360份
  for (let i = startLon; i <= 360 + startLon; i++) {
    // 计算每一经度对应的经纬度
    const lon = i
    const lat = Math.atan(tanA * Math.sin(degreeUnit * lon + y)) * radianUnit
    track.push({
      position: Cartesian3.fromDegrees(lon, lat, height),
      time: JulianDate.addSeconds(startTime, i, new JulianDate()),
    });
  }
  return track;
}
