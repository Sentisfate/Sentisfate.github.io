特殊数组的特征值

2022-09-12 04:00

题目描述

解法1 排序+枚举

思路：

将数组按降序排序，此时，对于第x个元素。如果满足

时，x为该数组的特征值。

临界条件： :::info 对于数组的第一个元素，如果

则说明数组中有0个元素大于等于0，0为数组的特征值。同理对于最后一个元素，如果

len为特征值 :::

/**
 * @param {number[]} nums
 * @return {number}
 */
var specialArray = function(nums) {
    //排序
    nums.sort((a,b)=>b-a)
    //nums[x-1]>=x
    //nuums[x]<x
    if(nums[0]<0){
        return 0
    }
    let len=nums.length
    for(let x=1;x<len;x++){
        if(nums[x-1]>=x&&nums[x]<x){
            return x
        }
    }
    if(nums[len-1]>=len){
        return len
    }
    return -1
};

解法2.排序+二分

/**
 * @param {number[]} nums
 * @return {number}
 */
var specialArray = function(nums) {
    //排序
    nums.sort((a,b)=>b-a)
    //nums[x-1]>=x
    //nuums[x]<x
    let len=nums.length
    //记录查找区间的上限
    let left=0
    let right=len-1
    //x为第��个，因此要加1
    let x=Math.floor((right+left)/2)+1
    while(right>=left){
        if(nums[x-1]>=x&&(x==len||nums[x]<x)){
            return x
        }
        if(nums[x-1]>=x){
            //第x个的索引加1为x
            left=x
        }else{
            right=x-2
        }
        x=Math.floor((right+left)/2)+1
    }
    return -1
};
