webgl在gpu上的工作原理基本分为2部分：

  • 将顶点转换到裁剪空间坐标
  • 基于第一部分的结果绘制像素点
var primitiveType = gl.TRIANGLES;
var offset = 0;
var count = 9;
gl.drawArrays(primitiveType, offset, count);

上面这段代码的过程大致如下：

顶点着色器会依次处理每个顶点，根据drawArrays指定的次数，顶点着色器也会相应的运行对应次数，再依据绘制类型，webgl决定何时去画什么样的图形。绘制不同顶点颜色的三角形：

attribute vec2 a_position;
uniform mat3 u_m;
// 定义一个可变颜色，每个顶点一个颜色，将会被传递给片段着色器
varying vec4 v_color;      
void main() {
  vec3 position=vec3(a_position.x,a_position.y,1.0);
  vec3 result=u_m*position;
  gl_Position = vec4(result.x,result.y,0,1);
  // 将裁剪空间转为颜色空间
  v_color=gl_Position*0.5+0.5;
}

precision mediump float;
// 定义一个同名的可变变量来接收从顶点着色器传递过来的数据
varying vec4 v_color; 
void main(){
  gl_FragColor=v_color;
}

      const center = {

      // 确认 WebGL 支持性

      // 创建着色器的方法
      function createShader(gl, type, source) {
        const shader = gl.createShader(type);
        gl.shaderSource(shader, source); //为着色器对象提供数据源，即我们写的着色器代码
        gl.compileShader(shader); //编译着色器
        const success = gl.getShaderParameter(shader, gl.COMPILE_STATUS);
        // 成功编译后返回着色器对象
        if (success) return shader;
      }
      // 计算变换矩阵
      function getMat3(w, h, x, y) {
        return [
          2 / w,
          0.0,
          0,
          0.0,
          -2 / h,
          0,
          (-2.0 * x) / w,
          (2.0 * y) / h,
          1.0,
        ];
      }
      const vsSource = `
      attribute vec2 a_position;
      uniform mat3 u_m;
      // 定义一个可变颜色，每个顶点一个颜色，将会被传递给片段着色器
      varying vec4 v_color;      
      void main() {
        vec3 position=vec3(a_position.x,a_position.y,1.0);
        vec3 result=u_m*position;
        gl_Position = vec4(result.x,result.y,0,1);
        // 将裁剪空间转为颜色空间
        v_color=gl_Position*0.5+0.5;
      }
      `;
      const fsSource = `
      precision mediump float;
      // 定义一个同名的可变变量来接收从顶点着色器传递过来的数据
      varying vec4 v_color; 
      void main(){
        gl_FragColor=v_color;
      }
      `;
      // 创建着色器
      const vertexShader = createShader(gl, gl.VERTEX_SHADER, vsSource);
      const fragmentShader = createShader(gl, gl.FRAGMENT_SHADER, fsSource);
      // 将着色器链接到一个program着色程序
      function createProgram(gl, vertexShader, fragmentShader) {
        const program = gl.createProgram();
        gl.attachShader(program, vertexShader);
        gl.attachShader(program, fragmentShader);
        gl.linkProgram(program);
        const success = gl.getProgramParameter(program, gl.LINK_STATUS);
        if (success) return program;
        // 未链接成功，则打印相关信息，并删除失败的program
        console.log(gl.getProgramInfoLog(program));
        gl.deleteProgram(program);
      }
      const program = createProgram(gl, vertexShader, fragmentShader);

      const positionAttributeLocation = gl.getAttribLocation(
        program,
        "a_position"
      );
      const u_m = gl.getUniformLocation(program, "u_m");

      gl.viewport(0, 0, gl.canvas.width, gl.canvas.height);
      gl.clearColor(0, 0, 0, 0.5); //设置清空颜色
      gl.clear(gl.COLOR_BUFFER_BIT); //清空画布
      gl.useProgram(program);
      gl.enableVertexAttribArray(positionAttributeLocation);

      const positionBuffer = gl.createBuffer();
      gl.bindBuffer(gl.ARRAY_BUFFER, positionBuffer);
      gl.bufferData(
        gl.ARRAY_BUFFER,
        new Float32Array([0, 0, 800, 600, 0, 600,0,0,800,600,800,0]),
        gl.STATIC_DRAW
      );

      // 设置全局变量
      gl.uniformMatrix3fv(
        u_m,
        false,
        new Float32Array(
          getMat3(gl.canvas.width, gl.canvas.height, center.x, center.y)
        )
      );

      // 配置顶点属性指针
      gl.vertexAttribPointer(
        positionAttributeLocation,
        2,
        gl.FLOAT,
        false,
        0,
        0
      );
      const primitiveType = gl.TRIANGLES;
      gl.drawArrays(primitiveType, 0, 6);
    </script>

varying 可变变量，用于从顶点着色器向片段着色器中传递数据

这里绘制了两个三角形，把裁剪空间铺满，由于采用的可变颜色是通过裁剪空间的顶点数据转换为颜色空间得到的，因此体现的是一个r-g（b分量为0.5，a分量为1）渐变颜色空间的信息。下面是空间大致四个分量的范围

只传递到了顶点颜色的值，中间颜色是如何产生的？

webgl，在光栅化三角形时，会将三个值进行插值，每一个像素在调用片段着色器时，varying可变量的值是一个插值。

通过属性传递颜色给着色器：

attribute vec2 a_position;
attribute vec4 a_color;
uniform mat3 u_m;
// 定义一个可变颜色，每个顶点一个颜色，将会被传递给片段着色器
varying vec4 v_color;      
void main() {
  vec3 position=vec3(a_position.x,a_position.y,1.0);
  vec3 result=u_m*position;
  gl_Position = vec4(result.x,result.y,0,1);
  // 将裁剪空间转为颜色空间
  v_color=a_color;
}

precision mediump float;
// 定义一个同名的可变变量来接收从顶点着色器传递过来的数据
varying vec4 v_color; 
void main(){
  gl_FragColor=v_color;
}
