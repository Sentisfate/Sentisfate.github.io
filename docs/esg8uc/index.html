const map = new mapboxgl.Map({
    container: 'map', // container ID 地图容器的id
    style: 'mapbox://styles/mapbox/satellite-streets-v11', // style URL 地图的样式，街道，卫星...
    center: [114, 30.6], // starting position [lng, lat]地图中心经纬度
    zoom: 1, // starting zoom  地图初始缩放级别
    projection: 'globe' // display the map as a 3D globe  2D，3D
});
map.on('style.load', () => {
    map.setFog({
        'range': [-1, 2],
        'horizon-blend': 0.3,
        // 大气层颜色
        'color': 'rgba(0,0,0,0.9)',
        // 大气层之上的颜色
        'high-color': 'skyblue',
        'space-color': 'black',
        'star-intensity': 0.0
    }); // Set the default atmosphere style  设置大气样式
    
});
// fullscreen control
map.addControl(new mapboxgl.FullscreenControl({ container: document.querySelector('body') }));

  1. 地球的旋转 **map.easeTo({center,duration:1000,easing:n=>n})**
// 标识用户是否点击
let isUserInterface = false
// 转 map.easeTo({center,duration:1000,easing:n=>n})
function rotate() {
    let zoom = map.getZoom()
    if (zoom > 2) {
        return
    }
    if (isUserInterface) {
        return
    }
    let center = map.getCenter();
    center.lng += 0.2
    // center.lat += 2
    map.easeTo({
        // 要转到的中心经纬度
        center,
        // 动画时长
        duration: 10,
        // 播放速度
        easing: n => 1 * n
    })
}
map.on('moveend', () => {
    rotate()
})
map.on('click', () => {
    isUserInterface = !isUserInterface
    if (!isUserInterface) {
        rotate()
    }
})

import mapboxgl from 'mapbox-gl';
import 'mapbox-gl/dist/mapbox-gl.css';
import { token } from '@/config';
export default {
  methods: {
    initMap() {
      mapboxgl.accessToken = token;
      const map = new mapboxgl.Map({
        container: 'map', // container ID
        style: 'mapbox://styles/mapbox/streets-v11', // style URL
        center: [110, 30], // starting position [lng, lat]
        zoom: 1, // starting zoom
        projection: 'globe', // display the map as a 3D globe
      });
      map.on('style.load', () => {
        map.setFog({}); // Set the default atmosphere style
      });
    },
  },
  mounted() {
    this.initMap();
  },
};
