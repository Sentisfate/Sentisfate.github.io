import './App.css';
import React, { Component } from 'react';

class App extends Component {
  // 找到this的方法
  // 1.使用箭头函数，箭头函数没有自己的this，会找上一级的因此能够拿到
  //   方法2 使用bind方法
  toClick2 = (str) => {
    console.log('我被点击了2');
    // 改变数据的方法
    this.setState({
      msg: str,
    });
  };
  toClick(str) {
    console.log('我被点击了');
    this.setState({
      msg: str,
    });
  }
  addList = () => {
    let listInfo = this.state.listInfo;
    listInfo.push('提莫');
    this.setState({
      listInfo: listInfo,
    });
  };
  state = {
    msg: '你好react',
    listInfo: ['蛮王', '剑圣', '赵信'],
    isShow: true,
  };
  toggleShow = () => {
    let isShow = this.state.isShow;
    this.setState({
      isShow: !isShow,
    });
  };
  render() {
    return (
      // class 需要使用className
      // style需要使用{{}}
      //   点击事件传参会自动执行
      <div>
        <div
          onClick={this.toClick.bind(this, 'canshu')}
          className="class1"
          style={{ color: 'red' }}
        >
          <h1>hello react {this.state.msg}</h1>
          {/* <img src={require('./img/下载.jfif')} alt="" srcset="" /> */}
          {/* 图片的引用需要使用require */}
        </div>
        {/* bind 方式传参不会自动执行，onclick=方法传参点击事件会自动执行  */}
        {/* <div onClick={this.toClick2('参数1')}>{this.state.msg}</div> */}
        {/*  箭头函数方式传参不会自动执行 */}
        <div
          onClick={() => {
            this.toClick2('22222');
          }}
        >
          {this.state.msg}
        </div>
        <div onClick={this.addList}>草丛{this.state.listInfo.length}剑客</div>
        {this.state.listInfo.map((item, index) => {
          return <div key={index}>{item}</div>;
        })}
        <div onClick={this.toggleShow}>切换隐藏</div>
        {this.state.isShow ? '显示' : '隐藏'}
      </div>
    );
  }
}

export default App;
