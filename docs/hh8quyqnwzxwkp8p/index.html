import React from 'react';
import Icon from '@ant-design/icons';
import './ToolBar.css';
//填入SVG组件即可
const hocIcon = (IconSvg) => {
  return (
    props = { style: { fontSize: '18px', color: 'red' }, className: 'icon' }
  ) => {
    return <Icon component={IconSvg} {...props} />;
  };
};

export default hocIcon;
