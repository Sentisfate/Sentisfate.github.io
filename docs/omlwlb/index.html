解法1.贪心

思路，希望能够一次（0次数组长度为1）跳完，不行的话，就两次，三次……上一次跳进最后一次的位置范围内。以此类推找到第一次的初始位置。证明：假设跳完需要**n**次。第**n**次起跳所在的位置是**i****n**，所在位置的元素值为**nums[in]**则满足**i****n****+nums[in]>=nums.length-1**** 如果要使n最小，则**in**要尽可能的小，即小于它的数无法一次跳到最后**i****n-1****+nums[in-1]>=in**如果第n-1次没能跳过**i****n**，则还需要额外一次跳跃，才能保证跳到的······依次类推，到第一次时，需要满足****nums[0]>=i2**

/**
 * @param {number[]} nums
 * @return {number}
 */
var jump = function(nums) {
    //贪心算法，希望能一次跳完，如果不能就找到能跳到最后的位置的最远的一个位置，并寻找下一次能跳到这里的点
    let end=nums.length-1
    let count=0
    let cur=end
    while(cur>0){
        for(let i=end-1;i>=0;i--){
            if(i+nums[i]>=end){
                cur=i
            }
        }
        end=cur
        count++
    }
    return count
};

解法2.挨个跳一遍，计算最小的次数思路，当从一个点位开始跳时，可以得到能跳到的最远的位置以及下一次可以跳的点的范围。遍历当前可跳的点位，找出可以到达的最远的位置索引，如果该位置大于等于��组的最后一个位置，则停止计数，否则，循环下一次可跳的范围（该范围为当前能跳到的最近到最远的位置）

/**
 * @param {number[]} nums
 * @return {number}
 */
var jump = function(nums) {
    if(nums.length==1){
        return 0
    }
    let end=nums.length-1
    let count=1
    let start=0
    let max=nums[0]
    while(max<end){
        //此次起跳的结束位置
        let lastMax=max
        //开始跳
        for(let i=start;i<=lastMax;i++){
            //获得跳的最远的位置
            max=Math.max(nums[i]+i,max)
        }
        //下一次起跳的开始位置为上一次的起跳最远位置加1，结束位置为当前能跳的最远的位置
        start=lastMax+1
        count++
    }
    return count
};
