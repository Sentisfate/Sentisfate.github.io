function inOrderTraversal2(root) {
  // 栈
  let nodesStack = [];
  // 节点指针
  let node = root;
  let vals = [];
  while (node || nodesStack.length > 0) {
    // 如果当前节点非空，就将节点指针移动到左子节点
    if (node) {
      nodesStack.push(node);
      node = node.left;
      continue;
    }
    // 当前节点为null时，就节点指针指向栈顶的节点(父节点)，并出栈
    node = nodesStack.pop()
    vals.push(node.val);
    // 指针移动至右子节点
    node = node.right
  }
  return vals;
}

解法2相较于前序中序遍历，由于后序遍历是左->右->根的顺序，需要判断当前的节点是否是左节点或者是右节点，如果是左节点，就遍历右节点，如果是右节点就遍历根节点。大致流程：

  • 访问根节点
  • 如果节点不为空，入栈，访问左子节点
  • 如果节点为空，回到栈顶的节点，判断，如果是从左子节点返回，就访问右子节点，如果是从右子节点返回，就将根节点出栈，并比较弹出的节点和栈顶的节点的关系，如果是左子节点，就移动至右子节点，如果是右子节点，就继续弹出栈顶，重复执行直至该节点非右子节点或者栈为空

判断条件：

  1. 判断当前节点是否为空
    1. 不为空，节点入栈，并访问左子节点
    2. 为空
      1. 判断当前结点与栈顶结点的关系
        1. 是左子节点，就访问栈顶结点的右子节点
        2. 是右子节点，就将栈顶结点出栈，并判断出栈结点与栈顶结点的关系
          1. 是左子节点，就访问右子节点，跳出当前循环，执行主循环
          2. 是右子节点，就继续出栈，执行这个循环，直至栈空或者非右子节点
function postOrderTraversal2(root) {
  let nodesStack = []
  let vals = []
  let isright = false
  while (root || nodesStack[0]) {
    if (root) {
      nodesStack.push(root)
      root = root.left
      isright = false
      continue
    }
    // 如果当前节点为空，判断当前节点与栈顶结点的关系，为左子结点就访问右子节点，为右子节点就访问根节点，并出栈
    // 是左子节点，访问右子节点
    if (!isright) {
      root = nodesStack[nodesStack.length - 1].right
      isright = true
      continue
    }
    while (nodesStack.length) {
      // 弹出栈顶的节点访问该节点的值
      root = nodesStack.pop()
      vals.push(root.val)
      if (nodesStack.length && nodesStack[nodesStack.length - 1].left == root) {
        // 访问右子节点
        root = nodesStack[nodesStack.length - 1].right
        isright = true
        break
      }
      // 是右子节点,回到栈顶结点
      root = nodesStack[nodesStack.length - 1]
    }
  }
  return vals
}
