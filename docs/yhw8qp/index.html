腐烂的橘子

2022-09-08 17:19

题目描述

解法1 广度优先遍历bfs

思路：

查找的是感染所有新鲜橘子的最小时间，也就是说每次查找完腐烂的橘子周围的橘子再遍历后面的橘子，初始网格中所有的腐烂的橘子可以看成是同级的。

临界条件： :::info

  1. 初始网格中没有新鲜橘子时，返回0
  2. 使用一个数组来存储同级的腐烂的橘子，每次遍历周围新鲜的橘子时，将其感染为2
  3. 当数组为空时，说明新鲜的橘子没有了 :::
/**
 * @param {number[][]} grid
 * @return {number}
 */
var orangesRotting = function(grid) {
    let rows=grid.length
    let cols=grid[0].length
    let queue=[]
    //先查找到所有腐烂的橘子的位置
    const getOrangeByState=(state)=>{
        for(let i=0;i<rows;i++){
            for(let j=0;j<cols;j++){
                if(grid[i][j]==state){
                    queue.push([i,j])
                }
            }
        }
    }
    const isFreshOrange=(arr,i,j)=>{
        if(i<0||i>=rows||j<0||j>=cols){
            return
        }
        if(grid[i][j]==1){
            grid[i][j]=2
            arr.push([i,j])
        }
    }
    //排除掉没有新鲜的橘子的情况
    getOrangeByState(1)
    if(queue.length==0){
        return 0
    }
    queue=[]
    //找到腐烂的橘子
    getOrangeByState(2)
    let count=0
    while(queue.length){
        let arr=[]
        for(let i=0;i<queue.length;i++){
            // 查找同级的腐烂的橘子
            let row=queue[i][0]
            let col=queue[i][1]
            //如果上方有新鲜的橘子的话，就入队，并将其变为腐烂的橘子
            isFreshOrange(arr,row-1,col)
            isFreshOrange(arr,row+1,col)
            isFreshOrange(arr,row,col-1)
            isFreshOrange(arr,row,col+1)
        }
        count++
        queue=null
        queue=arr
    }
    getOrangeByState(1)
    if(queue.length){
        return -1
    }
    return count-1
};
