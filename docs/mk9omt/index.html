解法1 双指针加数组

思路：

将链表的每个节点用一个数组存起来，用指针指向头尾节点，一次添加尾节点，头节点，并移动指针。

临界条件： :::info

其中

为头指针，默认为0，

为尾指针，默认为长度-1 :::

/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {void} Do not return anything, modify head in-place instead.
 */
var reorderList = function(head) {
    let node=head.next
    let nodes=[]
    while(node){
        nodes.push(node)
        node=node.next
    }
    // 标识是接头部还是接尾部,为true表示将尾部的节点移动到前面
    let flag=true
    let start=0
    let end=nodes.length-1
    while(end>=start){
        if(flag){
            head.next=nodes[end--]
            flag=false
        }else{
            head.next=nodes[start++]
            flag=true
        }
        head=head.next
        head.next=null
    }
};

解法2.寻找链表中点+逆序排列+合并链表

/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {void} Do not return anything, modify head in-place instead.
 */
var reorderList = function(head) {
    let node=head
    let middle=head
    let count=0
    //快指针指到倒数第二个节点时，慢指针指向中间节点的前一个，可以在此断开链表
    while(node.next){
        node=node.next
        count++
        if(count==2){
            middle=middle.next
            count=0
        }
    }
    let head2=middle.next
    if(!head2){
        return
    }
    let pre=head2
    let cur=pre.next
    middle.next=null
    head2.next=null
    //反转链表
    while(cur){
        pre=cur
        cur=cur.next
        pre.next=head2
        head2=pre
    }
    console.log(head)

    //合并两个链表
    while(head&&head2){
        let node1=head
        let node2=head2
        head2=head2.next
        head=head.next
        node1.next=node2
        node2.next=head
    }
    
};
