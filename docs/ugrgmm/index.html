最大同值路径

2022-09-02 06:17

题目描述

思路：对于一般路径而言，可以通过该路径穿过的最高节点将路径分为两部分，左路径和右路径，如：

本题中即可转换为求左最大同值有向路径和右最大同值有向路径之和。

/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var longestUnivaluePath = function(root) {
    // 大致思路，对于一般路径而言，可以将穿过某个根节点的路径以这个根节点分类为2部分，左和右，该路径就为左+右
    // 使用深度优先算法，遍历二叉树，当遍历到当前结点时，穿过该节点的最长同值(以下均省略同值)路径为左最长有向路径加右最长有向路径

    // 最长路径
    let max=0
    // 递归遍历每个结点，拿到它的最长有向路径
    const dfs=(root)=>{
        if(root==null){
            return 0
        }
        // 左子节点的最长有向路径为dfs(root.left)
        let left=dfs(root.left)
        // 右子节点的最长有向路径
        let right=dfs(root.right)
        // 当前节点的左右最长有向路径
        
        let curLeft=curRight=0
        // 如果当前结点与左子节点相同，当前结点的左最长有向路径为左子节点的最长有向路径加1
        if(root.left&&root.left.val==root.val){
            curLeft=left+1
        }
        if(root.right&&root.right.val==root.val){
            curRight=right+1
        }
        //获得当前的最长同值有向路径
        max=Math.max(max,curLeft+curRight)
        // 返回当前节点的最长有向路径
        return Math.max(curRight,curLeft)
    }
    dfs(root)
    return max
};
