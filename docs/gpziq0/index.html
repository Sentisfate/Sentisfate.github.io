装饰器

装饰器是用@使用的，@浏览器识别不出语法，属于ES7的语法，需要使用babel将其翻译成浏览器可以识别的ES5语法

装饰器:对于类来说，就是增加了类的方法和属性，而没有改变类本身。即在类的原型对象上添加方法和属性

import React, { Component } from 'react';

// 装饰器
function addFun(ele) {
  ele.prototype.study = function () {
    console.log('study');
  };
  return ele;
}
class Home extends Component {
  render() {
    console.log(this);
    return (
      <div>
        <h1>主页</h1>
      </div>
    );
  }
}
const Home2 = addFun(Home);
export default Home2;
//或者这样写 export default addFun(Home)
//或者这样写 addFun(Home) export default Home2;

简写：在class前面使用 @方法名 来对下面的类进行方法的调用

import React, { Component } from 'react';

// 装饰器
function addFun(ele) {
  ele.prototype.study = function () {
    console.log('study');
  };
  return ele;
}
@addFun
class Home extends Component {
  render() {
    console.log(this);
    return (
      <div>
        <h1>主页</h1>
      </div>
    );
  }
}
// const Home2 = addFun(Home);
export default Home;

设计思想

利用定义在class上的属性，其被所有实例共有，从而在各个组件中引入实例之后，可以使用该属性。

class集成

mobx6以下的用法。

import { observable, action } from 'mobx';

// 状态管理的仓库
// mobx6以下的用法
class Store {
  @observable count = 0; //定义一个可观察的状态
  @action changeCount() {
    this.count += 1;
  }
}

export default new Store();

6及以上的方法

import { makeAutoObservable } from 'mobx';

// 状态管理的仓库
// mobx6以下的用法
class Store {
  constructor() {
    // 自动观察所有的状态值，相当于监听的效果
    makeAutoObservable(this);
  }
  count = 0;
  changeCount() {
    this.count++;
  }
}
export default new Store();

配置main.jsx,使用provider提供store

import React from 'react';
import ReactDOM from 'react-dom/client';
import App from './App';
import { Provider } from 'mobx-react';
import store from './store';
ReactDOM.createRoot(document.getElementById('root')).render(
  <Provider store={store}>
    <App />
  </Provider>
);

在组件中使用store中的数据。使用装饰器**@observer**进行组件的自动刷新

import React, { Component } from 'react';
import { Link } from 'react-router-dom';
import store from '../store';
import { observer } from 'mobx-react';

@observer
class ClassEle extends Component {
  changeCount = () => {
    store.changeCount();
    // 传一个空对象，会引起组件的刷新
    // this.setState({});
    console.log(store.count);
  };
  render() {
    console.log(this);
    console.log(store);
    return (
      <div>
        <h1>类组件页面</h1>
        <div>
          <button>
            {' '}
            <Link to={'/fun'}>go fun</Link>
          </button>
        </div>
        <button onClick={this.changeCount}>计数器：{store.count}</button>
      </div>
    );
  }
}

export default ClassEle;

函数式组件的使用 无法给函数添加装饰，要采用调用的形式**export default observer(FunEle);**

import { Link } from 'react-router-dom';
import store from '../store';
import { observer } from 'mobx-react';
// @observer
function FunEle(props) {
  return (
    <div>
      <button
        onClick={() => {
          store.changeCount();
        }}
        >
        {store.count}
      </button>
    </div>
    </div>
  );
}
export default observer(FunEle);
