二进制矩阵中的特殊位置

2022-09-04 04:28

题目描述：

解法1

思路1.找到所有1的位置，并将行列号唯一的1的个数计算出来。

/**
 * @param {number[][]} mat
 * @return {number}
 */
var numSpecial = function(mat) {
    // 思路：特殊位置处为1，其所在行列均为0,1的行列号都不相等，才为特殊值
    // 找出所有的1
    let rows=mat.length
    let cols=mat[0].length
    let arr=[]
    for(let i=0;i<rows;i++){
        for(let j=0;j<cols;j++){
            if(mat[i][j]){
                arr.push([i,j])
            }
        }
    }
    console.log(arr)
    let count=0
    arr.forEach((item,index)=>{
        if(arr.findIndex((el,i)=>(el[0]==item[0]||el[1]==item[1])&&i!=index)==-1){
            count++
        }
    })
    return count
};
