寻找重复的子树

2022-09-05 04:05

题目描述

思路：将二叉树序列化，并比较是否有重复的字符串。获得所有的子树序列化后的字符串存入哈希表，来得到重复的子串。如何序列化？通常有两种方式，一种序列化为深度优先，另一种为广度优先。

  • 深度优先可以通过前序遍历或后序遍历来完成，要保留空节点，才能唯一确定一个树
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode[]}
 */
var findDuplicateSubtrees = function(root) {
    let res=[]
    let map=new Map()
    const dfs=(root)=>{
    		let serial=''
        //用#标识空节点
        if(!root){
            return '#'
        }
        serial+=root.val
        //使用|来划分根节点和左右节点，因为左右节点如果有一个为空就会是#可以不用考虑拼接后的相同问题，但是根节点与子节点拼接后可能存在不唯一，因此要分割开。如根节点为1，左节点为11和根节点为11，左节点为1
        serial+='|'
        //将当前节点与左子树序列进行拼接
        serial+=dfs(root.left)
        //与右子树序列进行拼接
        serial+=dfs(root.right)
        if(map.has(serial)){
            let c=map.get(serial)
            if(c==1){
                res.push(root)
            }
            map.set(serial,map.get(serial)+1)
        }else{
            map.set(serial,1)
        }
      	//返回以当前节点为根节点的树的序列
        return serial
    }
    dfs(root)
    return res
};
